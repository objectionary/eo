5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import java.nio.file.FileSystems;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.nio.file.Paths;
12  import java.util.Collection;
13  import java.util.LinkedList;
14  import java.util.List;
15  import java.util.stream.Collectors;
16  import java.util.stream.Stream;
17  import org.cactoos.list.ListEnvelope;
18  
19  /**
20   * Walk through files in a directory.
21   *
22   * @since 0.1
23   */
24  final class Walk extends ListEnvelope<Path> {
25  
26      /**
27       * The home.
28       */
29      private final Path home;
30  
31      /**
32       * Ctor.
33       *
34       * @param dir The directory
35       */
36      Walk(final Path dir) {
37          this(dir, Walk.list(dir));
38      }
39  
40      /**
41       * Ctor.
42       *
43       * @param dir The directory
44       * @param list The list
45       */
46      private Walk(final Path dir, final List<Path> list) {
47          super(list);
48          this.home = dir;
49      }
50  
51      /**
52       * Includes this globs.
53       * @param globs List of them
54       * @return New Walk
55       */
56      Walk includes(final Collection<String> globs) {
57          return new Walk(
58              this.home,
59              this.stream()
60                  .filter(
61                      file -> globs.stream().anyMatch(
62                          glob -> this.matches(glob, file)
63                      )
64                  )
65                  .collect(Collectors.toList())
66          );
67      }
68  
69      /**
70       * Includes this globs.
71       * @param globs List of them
72       * @return New Walk
73       */
74      Walk excludes(final Collection<String> globs) {
75          return new Walk(
76              this.home,
77              this.stream()
78                  .filter(
79                      file -> globs.stream().noneMatch(
80                          glob -> this.matches(glob, file)
81                      )
82                  )
83                  .collect(Collectors.toList())
84          );
85      }
86  
87      /**
88       * List them all.
89       * @param dir The dir
90       * @return List
91       */
92      private static List<Path> list(final Path dir) {
93          try {
94              final List<Path> files = new LinkedList<>();
95              if (Files.exists(dir)) {
96                  files.addAll(Walk.regular(dir));
97              }
98              return files;
99          } catch (final IOException ex) {
100             throw new IllegalStateException(
101                 String.format("Can't read files in %s folder during a walk", dir),
102                 ex
103             );
104         }
105     }
106 
107     /**
108      * Get regular files from directory.
109      * @param dir The dir.
110      * @return Collection of files.
111      * @throws IOException If fails.
112      */
113     private static Collection<Path> regular(final Path dir) throws IOException {
114         try (Stream<Path> walk = Files.walk(dir)) {
115             return walk.filter(file -> !file.toFile().isDirectory())
116                 .collect(Collectors.toList());
117         }
118     }
119 
120     /**
121      * Create glob matcher from text.
122      * @param text The pattern, e.g. "**&#47;*.java"
123      * @param file The file to match
124      * @return Matcher
125      */
126     private boolean matches(final String text, final Path file) {
127         final Path rel = Paths.get(
128             file.toAbsolutePath().toString().substring(
129                 this.home.toAbsolutePath().toString().length() + 1
130             )
131         );
132         return FileSystems.getDefault().getPathMatcher(
133             String.format("glob:%s", text)
134         ).matches(rel);
135     }
136 }
