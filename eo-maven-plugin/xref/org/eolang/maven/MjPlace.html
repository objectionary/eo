5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import org.apache.maven.plugins.annotations.LifecyclePhase;
9   import org.apache.maven.plugins.annotations.Mojo;
10  
11  /**
12   * Take binary files from where {@link MjResolve} placed them and
13   * copy to the {@code target/classes} directory.
14   *
15   * <p>
16   *     Input directory is {@link MjResolve#DIR}.
17   *     Output directory is {@link MjPlace#targetDir}/classes.
18   * </p>
19   *
20   * @see <a href="https://news.eolang.org/2022-10-19-placed-catalog.html">Place catalog</a>
21   * @since 0.11
22   */
23  @Mojo(
24      name = "place",
25      defaultPhase = LifecyclePhase.PROCESS_SOURCES,
26      threadSafe = true
27  )
28  public final class MjPlace extends MjSafe {
29  
30      @Override
31      public void exec() throws IOException {
32          new Placing(
33              this.placedTojos,
34              this.targetDir.toPath().resolve(MjResolve.DIR),
35              this.classesDir.toPath(),
36              this.placeBinaries,
37              this.skipBinaries,
38              this.rewriteBinaries
39          ).exec();
40      }
41  }
