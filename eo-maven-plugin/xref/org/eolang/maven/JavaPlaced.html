5   package org.eolang.maven;
6   
7   import com.github.lombrozo.xnav.Xnav;
8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.nio.file.Paths;
12  import java.util.Arrays;
13  import org.cactoos.BiProc;
14  
15  /**
16   * Placed Java generated code.
17   * @since 0.56.7
18   */
19  final class JavaPlaced implements BiProc<Xnav, Boolean> {
20  
21      /**
22       * The footprint.
23       */
24      private final Footprint footprint;
25  
26      /**
27       * The target path.
28       */
29      private final Path target;
30  
31      /**
32       * Generated sources dir.
33       */
34      private final Path generated;
35  
36      /**
37       * Ctor.
38       * @param ftprnt The footprint
39       * @param tgt The target path
40       * @param gen Generated sources dir
41       */
42      JavaPlaced(final Footprint ftprnt, final Path tgt, final Path gen) {
43          this.footprint = ftprnt;
44          this.target = tgt;
45          this.generated = gen;
46      }
47  
48      @Override
49      public void exec(final Xnav clazz, final Boolean tests) throws IOException {
50          if (clazz.element("java").text().isPresent()) {
51              this.footprint.apply(Paths.get(""), this.target);
52          }
53          if (tests && JavaPlaced.testsPresent(clazz)) {
54              this.placeJavaTests(clazz);
55          }
56      }
57  
58      /**
59       * Place Java tests.
60       * @param clazz Transpiled Class
61       * @throws IOException If I/O fails
62       */
63      private void placeJavaTests(final Xnav clazz) throws IOException {
64          final String[] jparts = clazz.attribute("java-name").text().get().split("\\.");
65          final Path tests = this.generated.getParent().resolve(
66              "generated-test-sources"
67          );
68          final Path base = Arrays.stream(jparts, 0, jparts.length - 1)
69              .reduce(
70                  tests,
71                  Path::resolve,
72                  Path::resolve
73              );
74          final String origin = String.format("%sTest.java", jparts[jparts.length - 1]);
75          final Path resolved = base.resolve(origin);
76          final Path resulted;
77          final String content;
78          if (
79              Files.exists(
80                  tests.getParent().getParent().resolve("src").resolve("test")
81                      .resolve("java")
82                      .resolve(tests.relativize(resolved))
83              )
84          ) {
85              final String atomized = String.format(
86                  "%sEOAtomTest.java", jparts[jparts.length - 1]
87              );
88              resulted = base.resolve(atomized);
89              content = clazz.element("tests").text().get().replace(
90                  origin.replace(".java", ""), atomized.replace(".java", "")
91              );
92          } else {
93              resulted = resolved;
94              content = clazz.element("tests").text().get();
95          }
96          new Saved(content, resulted).value();
97      }
98  
99      /**
100      * Tests present?
101      * @param clazz Transpiled clazz
102      * @return True or False
103      */
104     private static boolean testsPresent(final Xnav clazz) {
105         return clazz.element("tests").text().map(s -> s.contains("@Test")).orElse(false);
106     }
107 }
