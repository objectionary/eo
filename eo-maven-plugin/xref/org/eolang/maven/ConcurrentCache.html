5   package org.eolang.maven;
6   
7   import java.nio.file.Path;
8   import java.util.concurrent.ConcurrentHashMap;
9   import java.util.concurrent.ConcurrentMap;
10  
11  /**
12   * Concurrent cache wrapper for Cache.
13   * Wrap {@link Cache} to make it thread-safe.
14   * @since 0.60
15   * @todo #4884:30min Use ReentrantLock instead of synchronized block.
16   *  This is the suggestion from qulice.
17   *  Dont' forget to remove the PMD suppression for AvoidSynchronizedStatement
18   *  after that.
19   */
20  @SuppressWarnings("PMD.AvoidSynchronizedStatement")
21  final class ConcurrentCache {
22  
23      /**
24       * Original cache.
25       */
26      private final Cache original;
27  
28      /**
29       * Locks for each cache entry.
30       */
31      private final ConcurrentMap<? super Path, Object> locks;
32  
33      /**
34       * Ctor.
35       * @param original Original cache
36       */
37      ConcurrentCache(final Cache original) {
38          this(original, new ConcurrentHashMap<>(0));
39      }
40  
41      /**
42       * Ctor.
43       * @param original Original cache
44       * @param locks Locks map
45       */
46      private ConcurrentCache(final Cache original, final ConcurrentMap<? super Path, Object> locks) {
47          this.original = original;
48          this.locks = locks;
49      }
50  
51      /**
52       * Check cache and apply compilation if needed.
53       * @param source From file
54       * @param target To file
55       * @param tail Tail path in cache
56       */
57      void apply(final Path source, final Path target, final Path tail) {
58          final Object loc = this.locks.computeIfAbsent(tail.normalize(), k -> new Object());
59          synchronized (loc) {
60              this.original.apply(source, target, tail.normalize());
61          }
62      }
63  }
