25  /**
26   * Find all required runtime dependencies, download
27   * them from Maven Central, unpack and place to the {@code target/eo}
28   * directory.
29   *
30   * <p>The motivation for this mojo is simple: Maven doesn't have
31   * a mechanism for adding .JAR files to transpile/test classpath in
32   * runtime.</p>
33   *
34   * @since 0.1
35   */
36  @Mojo(
37      name = "resolve",
38      defaultPhase = LifecyclePhase.PROCESS_SOURCES,
39      threadSafe = true
40  )
41  public final class MjResolve extends MjSafe {
42  
43      /**
44       * The directory where to resolve to.
45       */
46      static final String DIR = "4-resolve";
47  
48      /**
49       * Transitive dependency extractor. It's a strategy pattern for extracting transitive
50       * dependencies for a particular artifact.
51       *
52       * @checkstyle MemberNameCheck (7 lines)
53       */
54      @SuppressWarnings({"PMD.ImmutableField", "PMD.LongVariable"})
55      private Func<Dep, Dependencies> transitiveStrategy = dependency -> new DpsDepgraph(
56          this.project,
57          this.session,
58          this.manager,
59          this.targetDir.toPath().resolve(MjResolve.DIR).resolve("dependencies-info"),
60          dependency
61      );
62  
63      /**
64       * Resolve default JNA dependency or not.
65       *
66       * @checkstyle MemberNameCheck (7 lines)
67       */
68      @SuppressWarnings("PMD.ImmutableField")
69      private boolean resolveJna = true;
70  
71      /**
72       * Resolve dependencies in central or not.
73       * @checkstyle MemberNameCheck (7 lines)
74       */
75      @SuppressWarnings("PMD.ImmutableField")
76      private boolean resolveInCentral = true;
77  
78      @Override
79      public void exec() throws IOException {
80          final Collection<Dep> deps = this.deps();
81          if (deps.isEmpty()) {
82              Logger.info(this, "No new dependencies unpacked");
83          } else {
84              final Path target = this.targetDir.toPath().resolve(MjResolve.DIR);
85              new Threaded<>(
86                  deps,
87                  dep -> this.resolved(dep, target)
88              ).total();
89              Logger.info(
90                  this,
91                  "New %d dependenc(ies) unpacked to %[file]s: %s",
92                  deps.size(), target,
93                  new Joined(", ", new Mapped<>(Dep::toString, deps))
94              );
95          }
96      }
97  
98      /**
99       * Checks if dependency is runtime.
100      * @param dep Dependency
101      * @return True if runtime.
102      */
103     @SuppressWarnings("PMD.ProhibitPublicStaticMethods")
104     public static boolean isRuntime(final Dependency dep) {
105         return "org.eolang".equals(dep.getGroupId())
106             && "eo-runtime".equals(dep.getArtifactId());
107     }
108 
109     /**
110      * Resolved dependency.
111      * @param dep Dependency
112      * @param target Target path
113      * @return Amount of resolved dependencies
114      * @throws IOException If fails to resolve
115      */
116     private int resolved(final Dep dep, final Path target) throws IOException {
117         final String classifier;
118         final Dependency dependency = dep.get();
119         if (dependency.getClassifier() == null
120             || dependency.getClassifier().isEmpty()) {
121             classifier = "-";
122         } else {
123             classifier = dependency.getClassifier();
124         }
125         final Path dest = this.cleanPlace(
126             target
127                 .resolve(dependency.getGroupId())
128                 .resolve(dependency.getArtifactId())
129                 .resolve(classifier),
130             dependency.getVersion()
131         );
132         final int total;
133         if (Files.exists(dest)) {
134             Logger.debug(
135                 this,
136                 "Dependency %s already resolved and exists in %[file]s",
137                 dep, dest
138             );
139             total = 0;
140         } else {
141             this.central.accept(dependency, dest);
142             final int files = new Walk(dest).size();
143             if (files == 0) {
144                 Logger.warn(this, "No new files after unpacking of %s!", dep);
145             } else {
146                 Logger.info(
147                     this, "Found %d new file(s) (%d MB) after unpacking of %s",
148                     files, MjResolve.folderSizeInMb(dest), dep
149                 );
150             }
151             total = 1;
152         }
153         return total;
154     }

156     /**
157      * Returns directory where the files should be unpacked,
158      * making sure there are no old files around.
159      *
160      * <p>Say, on the first run of "resolve" binary files from some JAR
161      * get unpacked. Then, the user changes the version of the JAR
162      * in the "pom.xml" (or in some .eo file) and runs the build again.
163      * We don't want the old binary files to stay in the "5-resolve"
164      * directory, because they are outdated. We want them to be removed
165      * and only the new files from the right version of the JAR to be there.
166      * This method does exactly this: it removes old files and doesn't
167      * touches new ones.</p>
168      *
169      * @param dir The dir
170      * @param version Version of dependency
171      * @return Full path
172      * @throws IOException If fails
173      */
174     private Path cleanPlace(final Path dir, final String version) throws IOException {
175         final File[] subs = dir.toFile().listFiles();
176         if (subs != null) {
177             for (final File sub : subs) {
178                 final String base = sub.getName();
179                 Logger.info(this, "Base if %s", base);
180                 if (base.equals(version)) {
181                     continue;
182                 }
183                 final Path bad = dir.resolve(base);
184                 try (Stream<Path> walk = Files.walk(bad)) {
185                     walk
186                         .map(Path::toFile)
187                         .sorted(Comparator.reverseOrder())
188                         .forEach(File::delete);
189                 }
190                 Logger.info(
191                     this,
192                     "Directory %[file]s deleted because it contained wrong version files (not %s)",
193                     bad, version
194                 );
195             }
196         }
197         return dir.resolve(version);
198     }

200     /**
201      * Find all deps for all Tojos.
202      *
203      * @return List of them
204      */
205     private Collection<Dep> deps() {
206         Dependencies deps = new DpsDefault(
207             this.scopedTojos(), this.discoverSelf, this.skipZeroVersions, this.resolveJna
208         );
209         if (this.ignoreRuntime) {
210             Logger.info(this, "Runtime dependency is ignored because eo:ignoreRuntime=TRUE");
211             deps = new DpsWithoutRuntime(deps);
212         } else {
213             final Optional<Dependency> runtime = this.runtimeDependencyFromPom();
214             if (runtime.isPresent()) {
215                 deps = new DpsWithRuntime(deps, new Dep(runtime.get()));
216                 Logger.info(
217                     this,
218                     "Runtime dependency added from pom with version: %s",
219                     runtime.get().getVersion()
220                 );
221             } else {
222                 if (this.resolveInCentral) {
223                     deps = new DpsWithRuntime(deps);
224                 } else {
225                     deps = new DpsOfflineRuntime(deps);
226                 }
227             }
228         }
229         if (!this.ignoreVersionConflicts) {
230             deps = new DpsUniquelyVersioned(deps);
231         }
232         if (!this.ignoreTransitive) {
233             deps = new DpsEachWithoutTransitive(deps, this.transitiveStrategy);
234         }
235         return new SetOf<>(deps)
236             .stream()
237             .sorted()
238             .distinct()
239             .collect(Collectors.toList());
240     }
241 
242     /**
243      * Runtime dependency from pom.xml.
244      *
245      * @return Dependency if found.
246      */
247     private Optional<Dependency> runtimeDependencyFromPom() {
248         final Optional<Dependency> res;
249         if (this.project == null) {
250             res = Optional.empty();
251         } else {
252             res = this.project
253                 .getDependencies()
254                 .stream()
255                 .filter(MjResolve::isRuntime)
256                 .findFirst();
257         }
258         return res;
259     }

261     /**
262      * Folder size in megabytes.
263      * @param path Folder
264      * @return Folder size
265      * @throws IOException if I/O fails
266      */
267     private static long folderSizeInMb(final Path path) throws IOException {
268         try (Stream<Path> paths = Files.walk(path)) {
269             return paths.filter(Files::isRegularFile).mapToLong(
270                 p -> {
271                     try {
272                         return Files.size(p);
273                     } catch (final IOException exception) {
274                         throw new IllegalStateException(
275                             String.format("Failed to calculate size in %s", p),
276                             exception
277                         );
278                     }
279                 }
280             ).sum() / 1024L / 1024L;
281         }
282     }
