5   package org.eolang.maven;
6   
7   import java.io.File;
8   import java.nio.file.Path;
9   import java.nio.file.Paths;
10  
11  /**
12   * Make the place for the object.
13   *
14   * @since 0.1
15   */
16  final class Place {
17      /**
18       * Name of the object.
19       */
20      private final String name;
21  
22      /**
23       * Ctor.
24       * @param obj The name of the object
25       */
26      Place(final String obj) {
27          this.name = obj;
28      }
29  
30      /**
31       * Make a full path.
32       * @param ext Extension
33       * @return Full path
34       */
35      Path make(final String ext) {
36          return this.make(Paths.get(""), ext);
37      }
38  
39      /**
40       * Make a full path.
41       * @param dir The dir
42       * @param ext The ext
43       * @return Full path
44       */
45      Path make(final Path dir, final String ext) {
46          final StringBuilder out = new StringBuilder();
47          out.append(this.name.replace(".", File.separator));
48          if (!ext.isEmpty()) {
49              out.append('.').append(ext);
50          }
51          return dir.resolve(out.toString());
52      }
53  
54  }
