5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import org.apache.maven.plugins.annotations.LifecyclePhase;
9   import org.apache.maven.plugins.annotations.Mojo;
10  
11  /**
12   * Pull all necessary EO XML files from Objectionary and parse them all.
13   *
14   * <p>
15   *     This goal combines {@link MjParse}, {@link MjProbe} and {@link MjPull}.
16   *     See their documentation to find out more details.
17   *     The {@link MjAssemble} runs these goals repeatedly until no new objects are
18   *     pulled or parsed in a cycle.
19   * </p>
20   *
21   * @since 0.1
22   */
23  @Mojo(
24      name = "assemble",
25      defaultPhase = LifecyclePhase.PROCESS_SOURCES,
26      threadSafe = true
27  )
28  public final class MjAssemble extends MjSafe {
29  
30      /**
31       * The intermediate representation extension.
32       */
33      static final String XMIR = "xmir";
34  
35      /**
36       * Source file extension.
37       */
38      static final String EO = "eo";
39  
40      @Override
41      public void exec() throws IOException {
42          this.assembling().exec();
43      }
44  }
