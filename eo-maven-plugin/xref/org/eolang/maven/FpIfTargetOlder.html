5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import org.cactoos.Func;
12  
13  /**
14   * Footprint that behaves like one of the given wrapped footprints depending on
15   * the result of comparison target and source in terms of last modified date.
16   * @since 0.41
17   */
18  public final class FpIfTargetOlder extends FpEnvelope {
19      /**
20       * Ctor.
21       * @param destination Function that modifies result target path
22       * @param first First wrapped footprint
23       * @param second Second wrapped footprint
24       */
25      FpIfTargetOlder(
26          final Func<Path, Path> destination, final Footprint first, final Footprint second
27      ) {
28          super(
29              new FpFork(
30                  (source, target) -> {
31                      final Path dest = destination.apply(target);
32                      final boolean older = FpIfTargetOlder.isAfter(dest, source);
33                      if (older) {
34                          Logger.debug(
35                              FpIfTargetOlder.class,
36                              "Target file %[file]s is older than source %[file]s",
37                              dest, source
38                          );
39                      } else {
40                          Logger.debug(
41                              FpIfTargetOlder.class,
42                              "Target file %[file]s is newer than source %[file]s",
43                              dest, source
44                          );
45                      }
46                      return older;
47                  },
48                  first,
49                  second
50              )
51          );
52      }
53  
54      /**
55       * Returns true if first given path is older in terms of last modified time.
56       * @param first First path to compare
57       * @param second Second path to compare
58       * @return True if first path is older that second path
59       * @throws IOException If fails to compare files
60       */
61      private static boolean isAfter(final Path first, final Path second) throws IOException {
62          return Files.getLastModifiedTime(first).toInstant().isAfter(
63              Files.getLastModifiedTime(second).toInstant()
64          );
65      }
66  }
