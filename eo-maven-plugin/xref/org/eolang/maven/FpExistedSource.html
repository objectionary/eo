5   package org.eolang.maven;
6   
7   /**
8    * Footprint throws exception if source file does not exist.
9    * @since 0.41
10   */
11  final class FpExistedSource extends FpEnvelope {
12  
13      /**
14       * Ctor.
15       * @param footprint Original footprint
16       */
17      FpExistedSource(final Footprint footprint) {
18          super(
19              (source, target) -> {
20                  if (!source.toFile().exists()) {
21                      throw new IllegalStateException(
22                          String.format("Source file %s does not exist", source)
23                      );
24                  }
25                  return footprint.apply(source, target);
26              }
27          );
28      }
29  }
