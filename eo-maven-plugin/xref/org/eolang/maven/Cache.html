5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import java.nio.file.Files;
9   import java.nio.file.Path;
10  import java.security.MessageDigest;
11  import java.security.NoSuchAlgorithmException;
12  import java.util.Base64;
13  import org.cactoos.Func;
14  import org.cactoos.func.UncheckedFunc;
15  
16  /**
17   * Simple cache mechanism.
18   * This class isn't thread-safe, use {@link ConcurrentCache} for concurrent scenarios.
19   * @since 0.60
20   */
21  final class Cache {
22  
23      /**
24       * Base cache directory.
25       */
26      private final Path base;
27  
28      /**
29       * Compilation function.
30       */
31      private final Func<Path, String> compilation;
32  
33      /**
34       * Constructor.
35       * @param path Cache path
36       * @param compilation Compilation function
37       */
38      Cache(final CachePath path, final Func<Path, String> compilation) {
39          this(path.get(), compilation);
40      }
41  
42      /**
43       * Ctor.
44       * @param base Base cache directory
45       * @param compilation Compilation function
46       */
47      Cache(final Path base, final Func<Path, String> compilation) {
48          this.base = base;
49          this.compilation = compilation;
50      }
51  
52      /**
53       * Check cache and apply compilation if needed.
54       * @param source From file
55       * @param target To file
56       * @param tail Tail path in cache
57       */
58      public void apply(final Path source, final Path target, final Path tail) {
59          try {
60              final String sha = Cache.sha(source);
61              final Path hash = this.hash(tail);
62              final Path cache = this.base.resolve(tail);
63              if (
64                  Files.notExists(hash)
65                      || Files.notExists(cache)
66                      || !Files.readString(hash).equals(sha)
67              ) {
68                  final String content = new UncheckedFunc<>(this.compilation).apply(source);
69                  new Saved(sha, this.hash(tail)).value();
70                  new Saved(content, cache).value();
71                  new Saved(content, target).value();
72              } else {
73                  new Saved(Files.readString(cache), target).value();
74              }
75          } catch (final IOException ioexception) {
76              throw new IllegalStateException(
77                  "Failed to perform an IO operation with cache",
78                  ioexception
79              );
80          } catch (final NoSuchAlgorithmException exception) {
81              throw new IllegalStateException("SHA-256 hashing algorithm isn't found", exception);
82          }
83      }
84  
85      /**
86       * Get hash file path for the given tail.
87       * @param tail Tail path
88       * @return Hash file path
89       */
90      private Path hash(final Path tail) {
91          final Path full = this.base.resolve(tail.normalize());
92          return full.getParent().resolve(String.format("%s.sha256", full.getFileName().toString()));
93      }
94  
95      /**
96       * Calculate SHA-256 hash of a file and return it as Base64 string.
97       * @param file File path
98       * @return Base64-encoded SHA-256 hash
99       * @throws NoSuchAlgorithmException If SHA-256 algorithm is not available
100      * @throws IOException If an I/O error occurs reading the file
101      */
102     private static String sha(final Path file) throws NoSuchAlgorithmException, IOException {
103         final MessageDigest digest = MessageDigest.getInstance("SHA-256");
104         try (var stream = Files.newInputStream(file)) {
105             final byte[] buffer = new byte[8192];
106             int read = stream.read(buffer);
107             while (read != -1) {
108                 digest.update(buffer, 0, read);
109                 read = stream.read(buffer);
110             }
111         }
112         final byte[] hash = digest.digest();
113         return Base64.getEncoder().encodeToString(hash);
114     }
115 }
