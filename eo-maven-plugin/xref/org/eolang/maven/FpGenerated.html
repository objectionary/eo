5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import java.nio.file.Path;
9   import org.cactoos.Func;
10  import org.cactoos.Input;
11  import org.cactoos.func.UncheckedFunc;
12  
13  /**
14   * Footprint that saves content generated from lambda to the target file.
15   * @since 0.41
16   */
17  final class FpGenerated implements Footprint {
18      /**
19       * Content function.
20       */
21      private final UncheckedFunc<Path, Input> content;
22  
23      /**
24       * Ctor.
25       * @param content Content as bytes
26       */
27      FpGenerated(final Func<Path, Input> content) {
28          this.content = new UncheckedFunc<>(content);
29      }
30  
31      @Override
32      public Path apply(final Path source, final Path target) throws IOException {
33          return new Saved(this.content.apply(source), target).value();
34      }
35  }
