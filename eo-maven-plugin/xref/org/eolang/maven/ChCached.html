5   package org.eolang.maven;
6   
7   import org.cactoos.scalar.Sticky;
8   import org.cactoos.scalar.Synced;
9   import org.cactoos.scalar.Unchecked;
10  
11  /**
12   * Cached commit hash.
13   *
14   * @since 0.28.11
15   */
16  final class ChCached implements CommitHash {
17  
18      /**
19       * Cache.
20       */
21      private final Unchecked<String> delegate;
22  
23      /**
24       * Default constructor.
25       *
26       * @param delegate Delegate
27       */
28      ChCached(final CommitHash delegate) {
29          this.delegate = new Unchecked<>(new Synced<>(new Sticky<>(delegate::value)));
30      }
31  
32      @Override
33      public String value() {
34          return this.delegate.value();
35      }
36  }
