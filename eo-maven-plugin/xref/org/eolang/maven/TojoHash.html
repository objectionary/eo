5   package org.eolang.maven;
6   
7   import java.util.function.Supplier;
8   
9   /**
10   * Optional tojo hash.
11   * Returns tojo hash if exists or empty string otherwise.
12   * @since 0.41
13   */
14  final class TojoHash implements Supplier<String> {
15      /**
16       * Tojo.
17       */
18      private final TjForeign tojo;
19  
20      /**
21       * Ctor.
22       * @param tjo Foreign tojo
23       */
24      TojoHash(final TjForeign tjo) {
25          this.tojo = tjo;
26      }
27  
28      @Override
29      public String get() {
30          final String hash;
31          if (this.tojo.hasHash()) {
32              hash = this.tojo.hash();
33          } else {
34              hash = "";
35          }
36          return hash;
37      }
38  }
