5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import com.jcabi.xml.XMLDocument;
9   import java.io.File;
10  import java.io.IOException;
11  import java.nio.file.Path;
12  import java.nio.file.Paths;
13  import org.apache.maven.plugins.annotations.LifecyclePhase;
14  import org.apache.maven.plugins.annotations.Mojo;
15  import org.apache.maven.plugins.annotations.Parameter;
16  import org.cactoos.text.TextOf;
17  import org.eolang.parser.Xmir;
18  
19  /**
20   * Print XMIR to EO.
21   * @since 0.33.0
22   */
23  @Mojo(
24      name = "print",
25      defaultPhase = LifecyclePhase.PROCESS_SOURCES,
26      threadSafe = true
27  )
28  public final class MjPrint extends MjSafe {
29      /**
30       * Directory with XMIR sources to print.
31       * @checkstyle MemberNameCheck (10 lines)
32       */
33      @Parameter(
34          property = "eo.printSourcesDir",
35          required = true,
36          defaultValue = "${project.basedir}/src/main/xmir"
37      )
38      private File printSourcesDir;
39  
40      /**
41       * Directory where printed EO files are placed.
42       * @checkstyle MemberNameCheck (10 lines)
43       */
44      @Parameter(
45          property = "eo.printOutputDir",
46          required = true,
47          defaultValue = "${project.build.directory}/generated-sources/eo"
48      )
49      private File printOutputDir;
50  
51      @Override
52      void exec() throws IOException {
53          final Path home = this.printOutputDir.toPath();
54          final int total = new Threaded<>(
55              new Walk(this.printSourcesDir.toPath()),
56              source -> {
57                  final Path relative = Paths.get(
58                      this.printSourcesDir.toPath().relativize(source).toString()
59                          .replace(".xmir", ".eo")
60                  );
61                  final String program = new Xmir(
62                      new XMLDocument(new TextOf(source).asString())
63                  ).toEO();
64                  new Saved(program, home.resolve(relative)).value();
65                  Logger.info(
66                      this,
67                      "Printed: %[file]s (%[size]s) => %[file]s (%[size]s)",
68                      source,
69                      source.toFile().length(),
70                      this.printOutputDir.toPath().resolve(relative),
71                      this.printOutputDir.toPath().resolve(relative).toFile().length()
72                  );
73                  return 1;
74              }
75          ).total();
76          if (total == 0) {
77              Logger.info(this, "No XMIR sources found");
78          } else {
79              Logger.info(this, "Printed %d XMIR sources into EO", total);
80          }
81      }
82  }
