5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.nio.file.Path;
9   import java.util.function.Supplier;
10  import org.cactoos.text.TextOf;
11  
12  /**
13   * Footprint that updates target from content function and updates cache from target.
14   * @since 0.41
15   */
16  final class FpUpdateBoth extends FpEnvelope {
17      /**
18       * Ctor.
19       * @param origin Original footprint that updates target from source
20       * @param cache Lazy path to cache
21       */
22      FpUpdateBoth(final Footprint origin, final Supplier<Path> cache) {
23          super(
24              (source, target) -> {
25                  final Path che = cache.get();
26                  Logger.debug(
27                      FpUpdateBoth.class,
28                      "Updating target %[file]s and cache %[file]s from source %[file]s",
29                      target, che, source
30                  );
31                  origin.apply(source, target);
32                  new Saved(new TextOf(target), che).value();
33                  return target;
34              }
35          );
36      }
37  }
