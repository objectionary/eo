5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.nio.file.Path;
9   import org.cactoos.BiFunc;
10  
11  /**
12   * Rewrite policy.
13   * @since 0.56.7
14   */
15  final class RewritePolicy implements BiFunc<Path, Path, Boolean> {
16  
17      /**
18       * Rewrite it?
19       */
20      private final boolean rewrite;
21  
22      /**
23       * Target directory.
24       */
25      private final Path target;
26  
27      /**
28       * Ctor.
29       * @param rwrte Rewrite?
30       * @param tgt Target directory
31       */
32      RewritePolicy(final boolean rwrte, final Path tgt) {
33          this.rewrite = rwrte;
34          this.target = tgt;
35      }
36  
37      @Override
38      public Boolean apply(final Path src, final Path result) throws Exception {
39          if (this.rewrite) {
40              Logger.debug(
41                  this,
42                  "Rewriting %[file]s because XMIR %[file]s was changed",
43                  result, this.target
44              );
45          }
46          return this.rewrite;
47      }
48  }
