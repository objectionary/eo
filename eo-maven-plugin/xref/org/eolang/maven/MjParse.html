5   package org.eolang.maven;
6   
7   import org.apache.maven.plugins.annotations.LifecyclePhase;
8   import org.apache.maven.plugins.annotations.Mojo;
9   import org.apache.maven.plugins.annotations.ResolutionScope;
10  
11  /**
12   * Parse EO to XML.
13   *
14   * <p>
15   *     This is the initial goal that parses all found EO sources to XMIRs.
16   *     You can read more about XMIR format
17   *     <a href="https://www.eolang.org/XMIR.html">here</a>
18   * </p>
19   * <p>
20   *    The goal scans all the EO sources registered in the foreign file catalog
21   *    (see {@link MjRegister} and {@link MjPull}) and then parses those that were not parsed
22   *    before (i.e. do not have XMIRs yet) to XMIR format.
23   *    The resulting XMIR files are stored in the {@link Parsing#DIR} directory.
24   * </p>
25   *
26   * @since 0.1
27   */
28  @Mojo(
29      name = "parse",
30      defaultPhase = LifecyclePhase.GENERATE_SOURCES,
31      threadSafe = true,
32      requiresDependencyResolution = ResolutionScope.COMPILE
33  )
34  public final class MjParse extends MjSafe {
35  
36      @Override
37      public void exec() {
38          new Parsing(
39              this.scopedTojos(),
40              this.targetDir.toPath(),
41              this.cache.toPath(),
42              this.cacheEnabled,
43              this.plugin.getVersion(),
44              this.sourcesDir.toPath()
45          ).exec();
46      }
47  }
