25  /**
26   * Parse EO to XML.
27   *
28   * <p>
29   *     This is the initial goal that parses all found EO sources to XMIRs.
30   *     You can read more about XMIR format
31   *     <a href="https://www.eolang.org/XMIR.html">here</a>
32   * </p>
33   * <p>
34   *    The goal scans all the EO sources registered in the foreign file catalog
35   *    (see {@link MjRegister} and {@link MjPull}) and then parses those that were not parsed
36   *    before (i.e. do not have XMIRs yet) to XMIR format.
37   *    The resulting XMIR files are stored in the {@link #DIR} directory.
38   * </p>
39   *
40   * @since 0.1
41   */
42  @Mojo(
43      name = "parse",
44      defaultPhase = LifecyclePhase.GENERATE_SOURCES,
45      threadSafe = true,
46      requiresDependencyResolution = ResolutionScope.COMPILE
47  )
48  public final class MjParse extends MjSafe {
49  
50      /**
51       * Zero version.
52       */
53      static final String ZERO = "0.0.0";
54  
55      /**
56       * The directory where to parse to.
57       */
58      static final String DIR = "1-parse";
59  
60      /**
61       * Subdirectory for parsed cache.
62       */
63      static final String CACHE = "parsed";
64  
65      @Override
66      public void exec() {
67          final long start = System.currentTimeMillis();
68          final Collection<TjForeign> tojos = this.scopedTojos().withSources();
69          final int total = new Threaded<>(
70              new Filtered<>(TjForeign::notParsed, tojos),
71              this::parsed
72          ).total();
73          if (0 == total) {
74              if (tojos.isEmpty()) {
75                  Logger.info(
76                      this,
77                      "No .eo sources registered, nothing to be parsed to XMIRs (maybe you forgot to execute the \"register\" goal?)"
78                  );
79              } else {
80                  Logger.info(
81                      this,
82                      "No new .eo sources out of %d parsed to XMIRs",
83                      tojos.size()
84                  );
85              }
86          } else {
87              Logger.info(
88                  this, "Parsed %d new .eo sources out of %d to XMIRs in %[ms]s",
89                  total, tojos.size(), System.currentTimeMillis() - start
90              );
91          }
92      }
93  
94      /**
95       * Parse EO file to XML.
96       *
97       * @param tojo The tojo
98       * @return Amount of parsed tojos
99       * @throws IOException If fails
100      */
101     private int parsed(final TjForeign tojo) throws Exception {
102         final Path source = tojo.source();
103         final String name = tojo.identifier();
104         final Path base = this.targetDir.toPath().resolve(MjParse.DIR);
105         final Path target = new Place(name).make(base, MjAssemble.XMIR);
106         final List<Node> refs = new ArrayList<>(1);
107         if (this.cacheEnabled) {
108             new ConcurrentCache(
109                 new Cache(
110                     new CachePath(
111                         this.cache.toPath().resolve(MjParse.CACHE),
112                         this.plugin.getVersion(),
113                         new TojoHash(tojo).get()
114                     ),
115                     src -> {
116                         final Node node = this.parsed(src, name);
117                         refs.add(node);
118                         return new XMLDocument(node).toString();
119                     }
120                 )
121             ).apply(source, target, base.relativize(target));
122         } else {
123             final Node node = this.parsed(source, name);
124             new Saved(new XMLDocument(node).toString(), target).value();
125             refs.add(node);
126         }
127         tojo.withXmir(target).withVersion(MjParse.version(target, refs));
128         final List<Xnav> errors = new Xnav(target)
129             .element("object")
130             .element("errors")
131             .elements(Filter.withName("error"))
132             .collect(Collectors.toList());
133         if (errors.isEmpty()) {
134             Logger.debug(this, "Parsed %[file]s to %[file]s", source, target);
135         } else {
136             for (final Xnav error : errors) {
137                 Logger.error(
138                     this,
139                     "Failed to parse '%[file]s:%s': %s",
140                     source,
141                     error.attribute("line").text().orElse("0"),
142                     error.text().orElse("")
143                 );
144             }
145         }
146         return 1;
147     }
148 
149     /**
150      * Source parsed to {@link Node}.
151      * @param source Relative source path
152      * @param identifier Name of the EO object as tojo identifier
153      * @return Parsed EO object as {@link Node}
154      * @throws IOException If fails to parse
155      */
156     private Node parsed(final Path source, final String identifier) throws IOException {
157         final EoSource.Xmir xmir = new EoSource(identifier, source).parsed();
158         Logger.debug(
159             MjParse.class,
160             "Parsed program '%s' from %[file]s:\n %s",
161             identifier, this.sourcesDir.toPath().relativize(source.toAbsolutePath()), xmir
162         );
163         if (xmir.broken()) {
164             new Saved(
165                 new TextOf(xmir.xml().toString()),
166                 this.targetDir.toPath().resolve(
167                     String.format("broken-%x.xmir", System.nanoTime())
168                 )
169             ).value();
170         }
171         return xmir.xml().inner();
172     }
173 
174     /**
175      * Tojo version.
176      * The version can be extracted from:
177      * 1. Parsed {@link Node} if EO object was parsed for the first time
178      * 2. XML document that was already parsed before
179      * @param target Path to result XML document
180      * @param parsed List with either one parsed {@link Node} or empty
181      * @return Tojo version
182      * @throws FileNotFoundException If XML document file does not exist
183      */
184     private static String version(
185         final Path target,
186         final List<Node> parsed
187     ) throws FileNotFoundException {
188         final Node node;
189         if (parsed.isEmpty()) {
190             node = new XMLDocument(target).inner();
191         } else {
192             node = parsed.get(0);
193         }
194         return new Xnav(node)
195             .element("object")
196             .element("metas")
197             .elements(
198                 Filter.all(
199                     Filter.withName("meta"),
200                     meta -> new Xnav(meta)
201                         .elements(
202                             Filter.all(
203                                 Filter.withName("head"),
204                                 head -> head.text().map("version"::equals).orElse(false)
205                             )
206                         )
207                         .findAny()
208                         .isPresent()
209                 )
210             )
211             .findFirst()
212             .map(meta -> meta.element("tail").text().orElse(MjParse.ZERO))
213             .orElse(MjParse.ZERO);
214     }
