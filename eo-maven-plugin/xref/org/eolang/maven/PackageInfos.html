5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.io.File;
9   import java.io.IOException;
10  import java.nio.file.Files;
11  import java.nio.file.Path;
12  import java.util.List;
13  import java.util.regex.Pattern;
14  import java.util.stream.Collectors;
15  
16  /**
17   * Package info classes.
18   * @since 0.60
19   */
20  final class PackageInfos {
21  
22      /**
23       * Pattern for replacing EO in package.
24       */
25      private static final Pattern PACKAGE = Pattern.compile("EO");
26  
27      /**
28       * Not allowed characters in package names.
29       */
30      private static final Pattern NOT_ALLOWED = Pattern.compile("[^a-zA-Z0-9_.$]");
31  
32      /**
33       * Pattern for package names beginning with a number.
34       */
35      private static final Pattern NUMBER_BEGINNING = Pattern.compile("^\\d.*");
36  
37      /**
38       * Directory where create package info files.
39       */
40      private final Path root;
41  
42      /**
43       * Constructor.
44       * @param root In which directory create files.
45       */
46      PackageInfos(final Path root) {
47          this.root = root;
48      }
49  
50      /**
51       * Create {@code package-info.java} files in all the directories under the {@link #root}.
52       * @return Amount of created files
53       * @throws IOException If fails to create a file
54       */
55      int create() throws IOException {
56          final int size;
57          if (Files.exists(this.root)) {
58              final List<Path> dirs = Files.walk(this.root)
59                  .filter(file -> Files.isDirectory(file) && !file.equals(this.root))
60                  .collect(Collectors.toList());
61              for (final Path dir : dirs) {
62                  final String pkg = this.root.relativize(dir).toString()
63                      .replace(File.separator, ".");
64                  final Path saved = new Saved(
65                      PackageInfos.content(pkg), dir.resolve("package-info.java")
66                  ).value();
67                  Logger.debug(this, "Created %s", saved);
68              }
69              size = dirs.size();
70          } else {
71              Logger.info(
72                  this,
73                  "No generated sources found, skipping package-info.java creation"
74              );
75              size = 0;
76          }
77          return size;
78      }
79  
80      private static String content(final String pkg) {
81          return String.join(
82              "\n",
83              "/**",
84              " * This file was auto-generated by eo-maven-plugin,",
85              " * don't modify it, all changes will be lost anyway.",
86              " */",
87              String.format(
88                  "// @org.eolang.XmirPackage(\"%s\")",
89                  PackageInfos.PACKAGE.matcher(pkg).replaceAll("")
90              ),
91              String.format("package %s;", PackageInfos.escaped(pkg))
92          );
93      }
94  
95      private static String escaped(final String pkg) {
96          String escaped = PackageInfos.NOT_ALLOWED.matcher(pkg).replaceAll("_");
97          if (PackageInfos.NUMBER_BEGINNING.matcher(escaped).matches()) {
98              escaped = String.format("_%s", escaped);
99          }
100         return escaped;
101     }
102 }
