5   package org.eolang.maven;
6   
7   import com.github.lombrozo.xnav.Xnav;
8   import com.jcabi.aspects.RetryOnFailure;
9   import com.jcabi.xml.XMLDocument;
10  import java.io.IOException;
11  import java.net.URL;
12  import java.util.Iterator;
13  import java.util.concurrent.TimeUnit;
14  import java.util.function.Supplier;
15  import org.apache.maven.model.Dependency;
16  import org.cactoos.list.ListOf;
17  import org.cactoos.scalar.Sticky;
18  import org.cactoos.scalar.Synced;
19  import org.cactoos.scalar.Unchecked;
20  
21  /**
22   * Add runtime dependency to the list of dependencies, if it is absent there.
23   *
24   * @since 0.28.11
25   */
26  final class DpsWithRuntime implements Dependencies {
27  
28      /**
29       * Dependency downloaded by HTTP from Maven Central.
30       */
31      private static final Unchecked<Dep> MAVEN_DEPENDENCY = DpsWithRuntime.mavenDependency();
32  
33      /**
34       * All dependencies.
35       */
36      private final Iterable<Dep> delegate;
37  
38      /**
39       * Supplier of the eo-runtime dependency.
40       */
41      private final Unchecked<Dep> supplied;
42  
43      /**
44       * Constructor.
45       *
46       * @param dlg Dependencies delegate.
47       */
48      DpsWithRuntime(final Iterable<Dep> dlg) {
49          this(dlg, DpsWithRuntime.MAVEN_DEPENDENCY);
50      }
51  
52      /**
53       * Constructor.
54       *
55       * @param dlg Dependencies delegate.
56       * @param sup Dependency.
57       */
58      DpsWithRuntime(
59          final Iterable<Dep> dlg,
60          final Dep sup
61      ) {
62          this(dlg, new Unchecked<>(() -> sup));
63      }
64  
65      /**
66       * The main constructor.
67       *
68       * @param dlg Dependencies delegate.
69       * @param sup Supplier of the eo-runtime dependency.
70       */
71      DpsWithRuntime(
72          final Iterable<Dep> dlg,
73          final Unchecked<Dep> sup
74      ) {
75          this.delegate = dlg;
76          this.supplied = sup;
77      }
78  
79      @Override
80      public Iterator<Dep> iterator() {
81          final ListOf<Dep> all = new ListOf<>(this.delegate);
82          if (all.stream().noneMatch(dep -> DpsWithRuntime.isRuntime(dep.get()))) {
83              all.add(this.supplied.value());
84          }
85          return all.iterator();
86      }
87  
88      /**
89       * Runtime dependency source from Maven Central.
90       *
91       * @return Runtime dependency from Maven Central.
92       */
93      @RetryOnFailure(delay = 1L, unit = TimeUnit.SECONDS)
94      private static Unchecked<Dep> mavenDependency() {
95          final String url = String.format(
96              "https://repo.maven.apache.org/maven2/%s/maven-metadata.xml",
97              "org/eolang/eo-runtime"
98          );
99          return DpsWithRuntime.dependency(
100             () -> {
101                 try {
102                     return new Xnav(new XMLDocument(new URL(url)).inner())
103                         .element("metadata")
104                         .element("versioning")
105                         .element("latest")
106                         .text()
107                         .get();
108                 } catch (final IOException ex) {
109                     throw new IllegalStateException(
110                         String.format("Can't get eo-runtime dependency by the URL: %s", url),
111                         ex
112                     );
113                 }
114             }
115         );
116     }
117 
118     /**
119      * Is it our runtime dep?
120      * @param other The dep
121      * @return TRUE if it is
122      */
123     private static boolean isRuntime(final Dependency other) {
124         return "org.eolang".equals(other.getGroupId())
125             && "eo-runtime".equals(other.getArtifactId());
126     }
127 
128     /**
129      * Runtime dependency source.
130      *
131      * @param version Version of eo-runtime
132      * @return Maven Dependency.
133      */
134     private static Unchecked<Dep> dependency(final Supplier<String> version) {
135         return new Unchecked<>(
136             new Synced<>(
137                 new Sticky<>(
138                     () -> new Dep()
139                         .withGroupId("org.eolang")
140                         .withArtifactId("eo-runtime")
141                         .withVersion(version.get())
142                         .withClassifier("")
143                 )
144             )
145         );
146     }
147 }
