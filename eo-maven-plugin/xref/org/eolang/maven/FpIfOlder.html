5   package org.eolang.maven;
6   
7   import java.nio.file.Path;
8   import org.cactoos.Func;
9   
10  /**
11   * Footprint that behaves as first given wrapped {@link Footprint}
12   * if provided target exists and older than source.
13   * Behaves as second given wrapped {@link Footprint} otherwise.
14   * @since 0.41
15   */
16  final class FpIfOlder extends FpEnvelope {
17      /**
18       * Ctor.
19       * @param first First wrapped footprint
20       * @param second Second wrapped footprint
21       */
22      FpIfOlder(final Footprint first, final Footprint second) {
23          this(target -> target, first, second);
24      }
25  
26      /**
27       * Ctor.
28       * @param destination Function that modifies result target path
29       * @param first First wrapped footprint
30       * @param second Second wrapped footprint
31       */
32      FpIfOlder(
33          final Func<Path, Path> destination, final Footprint first, final Footprint second
34      ) {
35          super(
36              new FpIfTargetExists(
37                  destination,
38                  new FpIfTargetOlder(
39                      destination,
40                      first,
41                      second
42                  ),
43                  second
44              )
45          );
46      }
47  }
