5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.util.function.Consumer;
9   import org.cactoos.Func;
10  import org.cactoos.experimental.Threads;
11  import org.cactoos.iterable.Mapped;
12  import org.cactoos.number.SumOf;
13  
14  /**
15   * Processes elements in multiple threads.
16   *
17   * <p>We use this class to process a large number of elements in parallel,
18   * using all available processors. It is used in most Mojos, where multiple
19   * files must be processed parallel.</p>
20   *
21   * @param <T> The type of the element to iterate
22   * @since 0.1
23   */
24  final class Threaded<T> {
25  
26      /**
27       * The sources.
28       */
29      private final Iterable<T> sources;
30  
31      /**
32       * The function.
33       */
34      private final Func<T, Integer> scalar;
35  
36      /**
37       * The logger.
38       */
39      private final Consumer<? super String> logger;
40  
41      /**
42       * Ctor.
43       * @param src The sources
44       * @param fun The function to run
45       */
46      Threaded(final Iterable<T> src, final Func<T, Integer> fun) {
47          this(src, fun, message -> Logger.error(Threaded.class, message));
48      }
49  
50      /**
51       * Ctor.
52       * @param src The sources
53       * @param fun The function to run
54       * @param log The logger
55       */
56      Threaded(
57          final Iterable<T> src, final Func<T, Integer> fun, final Consumer<? super String> log
58      ) {
59          this.sources = src;
60          this.scalar = fun;
61          this.logger = log;
62      }
63  
64      /**
65       * Exec them all and count.
66       * @return How many succeeded
67       */
68      @SuppressWarnings("PMD.AvoidCatchingGenericException")
69      int total() {
70          return new SumOf(
71              new Threads<>(
72                  Runtime.getRuntime().availableProcessors() * 2,
73                  new Mapped<>(
74                      tojo -> () -> {
75                          try {
76                              return this.scalar.apply(tojo);
77                              // @checkstyle IllegalCatchCheck (1 line)
78                          } catch (final Exception ex) {
79                              final String message = String.format(
80                                  "Failed to process \"%s\" (%s)",
81                                  tojo, tojo.getClass().getCanonicalName()
82                              );
83                              this.logger.accept(message);
84                              throw new IllegalStateException(message, ex);
85                          }
86                      },
87                      this.sources
88                  )
89              )
90          ).intValue();
91      }
92  }
