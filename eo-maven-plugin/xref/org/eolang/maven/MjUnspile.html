5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.io.File;
9   import java.io.IOException;
10  import java.nio.file.Files;
11  import java.nio.file.Path;
12  import java.util.ArrayList;
13  import java.util.Collection;
14  import java.util.List;
15  import java.util.Set;
16  import java.util.regex.Pattern;
17  import java.util.stream.Collectors;
18  import org.apache.maven.plugins.annotations.LifecyclePhase;
19  import org.apache.maven.plugins.annotations.Mojo;
20  
21  /**
22   * Goes through all .class files and deletes those that
23   * were created from autogenerated sources so these binaries
24   * are not got into result JAR.
25   *
26   * @since 0.1
27   */
28  @Mojo(
29      name = "unspile",
30      defaultPhase = LifecyclePhase.PREPARE_PACKAGE,
31      threadSafe = true
32  )
33  @SuppressWarnings("PMD.ImmutableField")
34  public final class MjUnspile extends MjSafe {
35      /**
36       * Pattern for matching paths ended with .class.
37       */
38      private static final Pattern JAVA = Pattern.compile("\\.java$");
39  
40      /**
41       * Inner auto generated classes.
42       * <p>These globs are used for inner classes which may look like:
43       * - EOorg/EOeolang/EOnumber$1$2$4.class
44       * - EOorg/EOeolang/EOsm/EOsocket$EOΦorgeolangsyssocketα0ρ.class</p>
45       */
46      private static final Collection<String> INNER = List.of(
47          "**/EO*$[1-9]*.class", "**/*$EOΦ*.class"
48      );
49  
50      @Override
51      public void exec() throws IOException {
52          final Walk classes = new Walk(this.classesDir.toPath());
53          if (classes.isEmpty()) {
54              Logger.warn(this, "No .class files in %[file]s, nothing to unspile", this.classesDir);
55          } else {
56              this.unspile(classes);
57          }
58      }
59  
60      /**
61       * Unspile classes.
62       * @param classes Collection of compiled classes
63       */
64      private void unspile(final Walk classes) {
65          final Path generated = this.generatedDir.toPath();
66          final Set<String> included = new Walk(generated)
67              .stream()
68              .map(
69                  path -> MjUnspile.JAVA.matcher(
70                      generated.relativize(path).toString()
71                  ).replaceAll(".class").replace(File.separatorChar, '/')
72              )
73              .collect(Collectors.toSet());
74          included.addAll(MjUnspile.INNER);
75          final Collection<Path> filtered = new ArrayList<>(
76              classes.excludes(this.keepBinaries).includes(included)
77          );
78          if (filtered.isEmpty()) {
79              Logger.info(
80                  this, "No .class files out of %d deleted in %[file]s",
81                  classes.size(), this.classesDir.toPath()
82              );
83          } else {
84              final int unspiled = new Threaded<>(
85                  filtered,
86                  path -> {
87                      final int deleted;
88                      if (Files.deleteIfExists(path)) {
89                          Logger.debug(
90                              this,
91                              "Deleted %[file]s since was compiled from %[file]s",
92                              path, generated
93                          );
94                          deleted = 1;
95                      } else {
96                          deleted = 0;
97                      }
98                      return deleted;
99                  }
100             ).total();
101             new EmptyDirectoriesIn(this.classesDir).clear();
102             Logger.info(
103                 this,
104                 "Deleted %d .class files in %[file]s",
105                 unspiled, this.classesDir.toPath()
106             );
107         }
108     }
109 }
