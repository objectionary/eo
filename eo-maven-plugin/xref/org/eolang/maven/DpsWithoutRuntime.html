5   package org.eolang.maven;
6   
7   import java.util.Iterator;
8   import org.cactoos.iterator.Filtered;
9   
10  /**
11   * Remove runtime dependency from the list of dependencies, if it is present there.
12   * Useful for the "eo-runtime" module compilation.
13   *
14   * @since 0.29
15   */
16  final class DpsWithoutRuntime implements Dependencies {
17  
18      /**
19       * All dependencies.
20       */
21      private final Dependencies delegate;
22  
23      /**
24       * Constructor.
25       * @param decoratee Dependencies delegate.
26       */
27      DpsWithoutRuntime(final Dependencies decoratee) {
28          this.delegate = decoratee;
29      }
30  
31      @Override
32      public Iterator<Dep> iterator() {
33          return new Filtered<>(
34              dep -> !"eo-runtime".equals(dep.get().getArtifactId()),
35              this.delegate.iterator()
36          );
37      }
38  }
