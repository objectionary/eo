5   package org.eolang.maven;
6   
7   import java.io.File;
8   import java.nio.file.Path;
9   import java.util.regex.Pattern;
10  
11  /**
12   * Make program name from a path.
13   *
14   * @since 0.1
15   */
16  final class Unplace {
17      /**
18       * Pattern to catch .eo files.
19       */
20      private static final Pattern EO = Pattern.compile(".eo$");
21  
22      /**
23       * The parent dir.
24       */
25      private final Path parent;
26  
27      /**
28       * Ctor.
29       * @param dir The name of the parent dir
30       */
31      Unplace(final File dir) {
32          this(dir.toPath());
33      }
34  
35      /**
36       * Ctor.
37       * @param dir The name of the parent dir
38       */
39      Unplace(final Path dir) {
40          this.parent = dir;
41      }
42  
43      /**
44       * Make a program name.
45       * @param file The file
46       * @return The name of the program
47       */
48      String make(final Path file) {
49          return Unplace.EO.matcher(
50              file.toString().substring(this.parent.toString().length() + 1)
51          ).replaceAll("").replace(File.separator, ".");
52      }
53  }
