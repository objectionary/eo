5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import java.nio.file.Path;
9   import java.util.function.Supplier;
10  import org.cactoos.BiFunc;
11  
12  /**
13   * Footprint after application with cache.
14   * @since 0.56.7
15   */
16  final class FpAppliedWithCache implements Footprint {
17  
18      /**
19       * Input.
20       */
21      private final Footprint input;
22  
23      /**
24       * Cache.
25       */
26      private final Supplier<Path> cache;
27  
28      /**
29       * Rewrite.
30       */
31      private final BiFunc<Path, Path, Boolean> rewrite;
32  
33      /**
34       * Cache enabled?
35       */
36      private final boolean caching;
37  
38      /**
39       * Ctor.
40       * @param ipt Input
41       * @param che Cache
42       * @param rwrte Rewrite
43       * @param enabled Cache enabled?
44       * @checkstyle ParameterNumberCheck (3 lines)
45       */
46      FpAppliedWithCache(
47          final Footprint ipt,
48          final Supplier<Path> che,
49          final BiFunc<Path, Path, Boolean> rwrte,
50          final boolean enabled
51      ) {
52          this.input = ipt;
53          this.cache = che;
54          this.rewrite = rwrte;
55          this.caching = enabled;
56      }
57  
58      @Override
59      public Path apply(final Path source, final Path target) throws IOException {
60          final Footprint both = new FpUpdateBoth(this.input, this.cache);
61          return new FpFork(
62              this.rewrite,
63              new FpFork(this.caching, both, this.input),
64              new FpIfTargetExists(
65                  new FpIgnore(),
66                  new FpFork(
67                      this.caching,
68                      new FpIfTargetExists(
69                          trgt -> this.cache.get(),
70                          new FpUpdateFromCache(this.cache),
71                          both
72                      ),
73                      this.input
74                  )
75              )
76          ).apply(source, target);
77      }
78  }
