5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.util.function.Supplier;
9   
10  /**
11   * Footprint that behaves like one of the given footprints depending on
12   * hash of provided cache.
13   * Similar to {@link FpFork} but the condition is based on hash.
14   * @since 0.41
15   * @checkstyle ParameterNumberCheck (100 lines)
16   */
17  final class FpIfReleased extends FpEnvelope {
18  
19      /**
20       * Ctor.
21       * @param hash Git hash
22       * @param first First footprint to use if a version is released and a hash is present
23       * @param second Second footprint to use if a version is not released or a hash is not present
24       */
25      FpIfReleased(
26          final String hash,
27          final Footprint first,
28          final Footprint second
29      ) {
30          this(() -> hash, first, second);
31      }
32  
33      /**
34       * Ctor.
35       * @param hash Git hash
36       * @param first First footprint to use if a version is released and a hash is present
37       * @param second Second footprint to use if a version is not released or a hash is not present
38       */
39      FpIfReleased(
40          final Supplier<String> hash,
41          final Footprint first,
42          final Footprint second
43      ) {
44          super(
45              new FpFork(
46                  (source, target) -> {
47                      final String hsh = hash.get();
48                      final boolean cacheable = !hsh.isEmpty();
49                      if (cacheable) {
50                          Logger.debug(
51                              FpIfReleased.class,
52                              "The hash '%s' is good, using cache for %[file]s",
53                              hsh, target
54                          );
55                      } else {
56                          Logger.debug(
57                              FpIfReleased.class,
58                              "The hash is absent, not using cache for %[file]s",
59                              target
60                          );
61                      }
62                      return cacheable;
63                  },
64                  first,
65                  second
66              )
67          );
68      }
69  }
