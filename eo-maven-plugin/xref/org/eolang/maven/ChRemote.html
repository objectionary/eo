5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import org.cactoos.Text;
9   
10  /**
11   * Hash of tag from objectionary.
12   *
13   * @since 0.26
14   */
15  final class ChRemote implements CommitHash {
16  
17      /**
18       * Cached text of hashes.
19       */
20      private static final Text CACHE = new CommitHashesText();
21  
22      /**
23       * Count of retries possible.
24       */
25      private static final Integer RETRIES = 3;
26  
27      /**
28       * Tag.
29       */
30      private final String tag;
31  
32      /**
33       * Constructor.
34       *
35       * @param tag Tag
36       */
37      ChRemote(final String tag) {
38          this.tag = tag;
39      }
40  
41      @Override
42      public String value() {
43          try {
44              final String sha = new ChText(
45                  ChRemote.CACHE::asString,
46                  this.tag,
47                  ChRemote.RETRIES
48              ).value();
49              Logger.debug(this, "Git sha of %s is %s", this.tag, sha);
50              return sha;
51          } catch (final ChText.NotFound ex) {
52              throw new IllegalArgumentException(
53                  String.format(
54                      "Tag '%s' doesn't exist or the list of all tags was not loaded correctly",
55                      this.tag
56                  ),
57                  ex
58              );
59          }
60      }
61  }
