5   package org.eolang.maven;
6   
7   import java.nio.file.Files;
8   import java.nio.file.Path;
9   import java.util.Arrays;
10  import org.cactoos.bytes.BytesOf;
11  import org.cactoos.bytes.Md5DigestOf;
12  import org.cactoos.bytes.UncheckedBytes;
13  import org.cactoos.io.InputOf;
14  
15  /**
16   * MD5 hash of a file (its content).
17   *
18   * @since 0.24
19   */
20  final class FileHash {
21  
22      /**
23       * The file.
24       */
25      private final Path file;
26  
27      /**
28       * Ctor.
29       * @param path The name of the file
30       */
31      FileHash(final Path path) {
32          this.file = path;
33      }
34  
35      @Override
36      public String toString() {
37          final String hash;
38          if (Files.exists(this.file)) {
39              hash = Arrays.toString(
40                  new UncheckedBytes(
41                      new Md5DigestOf(new InputOf(new BytesOf(this.file)))
42                  ).asBytes()
43              );
44          } else {
45              hash = "";
46          }
47          return hash;
48      }
49  }
