5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.io.IOException;
9   import java.nio.file.Path;
10  import java.util.function.Supplier;
11  import org.cactoos.text.TextOf;
12  
13  /**
14   * Footprint that updates target from cache.
15   * @since 0.41
16   */
17  final class FpUpdateFromCache implements Footprint {
18  
19      /**
20       * Cached file supplier.
21       */
22      private final Supplier<Path> cache;
23  
24      /**
25       * Ctor.
26       * @param cache Lazy path to cache
27       */
28      FpUpdateFromCache(final Supplier<Path> cache) {
29          this.cache = cache;
30      }
31  
32      @Override
33      public Path apply(final Path source, final Path target) throws IOException {
34          Logger.debug(
35              FpUpdateFromCache.class,
36              "Updating only target %[file]s from cache %[file]s",
37              target, source
38          );
39          return new Saved(new TextOf(this.cache.get()), target).value();
40      }
41  }
