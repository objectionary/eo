50      /**
51       * Whitespace pattern.
52       */
53      private static final Pattern WHITESPACE = Pattern.compile("\\s+");
54  
55      /**
56       * Constructor.
57       */
58      CommitHashesMap() {
59          this(new CommitHashesText()::asString);
60      }
61  
62      /**
63       * Ctor.
64       * @param table Commit hashes table as string.
65       */
66      private CommitHashesMap(final String table) {
67          this(() -> table);
68      }
69  
70      /**
71       * Ctor.
72       * @param table Commit hashes table.
73       */
74      private CommitHashesMap(final Scalar<String> table) {
75          super(CommitHashesMap.fromTable(table));
76      }
77  
78      /**
79       * Prestructor from hashes table.
80       * You can read more about prestructors and why they are needed right
81       * <a href="https://www.yegor256.com/2021/08/04/prestructors.html">here</a>
82       * @param table Commit hashes table as string value.
83       * @return Map of commit hashes.
84       */
85      private static Map<String, CommitHash> fromTable(final Scalar<String> table) {
86          return new MapOf<>(
87              new Mapped<>(
88                  rows -> {
89                      final String[] row = CommitHashesMap.WHITESPACE.split(rows.asString());
90                      return new MapEntry<>(
91                          row[1],
92                          new ChCached(
93                              new ChNarrow(
94                                  new CommitHash.ChConstant(row[0])
95                              )
96                          )
97                      );
98                  },
99                  new Split(table::value, "\n")
100             )
101         );
102     }
