5   package org.eolang.maven;
6   
7   import com.jcabi.manifests.Manifests;
8   import java.util.Collection;
9   import java.util.Iterator;
10  import org.cactoos.list.ListOf;
11  
12  /**
13   * Dependencies together with offline EO runtime dependency.
14   * @since 0.56.5
15   */
16  final class DpsOfflineRuntime implements Dependencies {
17  
18      /**
19       * EO current offline version.
20       */
21      private static final Dep EO_OFFLINE = new Dep().withGroupId("org.eolang")
22          .withArtifactId("eo-runtime")
23          .withVersion(Manifests.read("EO-Version"));
24  
25      /**
26       * All dependencies.
27       */
28      private final Iterable<Dep> all;
29  
30      /**
31       * Ctor.
32       * @param dlg All dependencies
33       */
34      DpsOfflineRuntime(final Iterable<Dep> dlg) {
35          this.all = dlg;
36      }
37  
38      @Override
39      public Iterator<Dep> iterator() {
40          final Collection<Dep> deps = new ListOf<>(this.all);
41          final boolean present = deps.stream().anyMatch(
42              dep -> dep.toString().startsWith("org.eolang:eo-runtime:")
43          );
44          if (!present) {
45              deps.add(DpsOfflineRuntime.EO_OFFLINE);
46          }
47          return deps.iterator();
48      }
49  }
