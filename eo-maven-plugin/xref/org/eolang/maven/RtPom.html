5   package org.eolang.maven;
6   
7   import java.util.Optional;
8   import org.apache.maven.model.Dependency;
9   import org.apache.maven.project.MavenProject;
10  import org.cactoos.Scalar;
11  
12  /**
13   * Runtime dependency from the project POM file.
14   * @since 0.62.0
15   */
16  final class RtPom implements Scalar<Dep> {
17  
18      /**
19       * Maven project.
20       */
21      private final MavenProject project;
22  
23      /**
24       * Ctor.
25       * @param mvn Maven project
26       */
27      RtPom(final MavenProject mvn) {
28          this.project = mvn;
29      }
30  
31      @Override
32      public Dep value() {
33          return this.runtimeFromPom()
34              .map(Dep::new).orElseThrow(
35                  () -> new IllegalStateException("Runtime dependency not found in pom.xml")
36              );
37      }
38  
39      /**
40       * Checks if runtime dependency is present in pom.xml.
41       * @return True if runtime dependency found
42       */
43      boolean isPresent() {
44          return this.runtimeFromPom().isPresent();
45      }
46  
47      /**
48       * Runtime dependency from pom.xml.
49       * @return Dependency if found
50       */
51      private Optional<Dependency> runtimeFromPom() {
52          final Optional<Dependency> res;
53          if (this.project == null) {
54              res = Optional.empty();
55          } else {
56              res = this.project
57                  .getDependencies()
58                  .stream()
59                  .filter(RtPom::isRuntime)
60                  .findFirst();
61          }
62          return res;
63      }
64  
65      /**
66       * Checks if dependency is the eo-runtime artifact.
67       * @param dep Dependency
68       * @return True if runtime
69       */
70      private static boolean isRuntime(final Dependency dep) {
71          return "org.eolang".equals(dep.getGroupId())
72              && "eo-runtime".equals(dep.getArtifactId());
73      }
74  }
