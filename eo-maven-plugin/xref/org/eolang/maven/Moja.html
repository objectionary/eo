5   package org.eolang.maven;
6   
7   import java.lang.reflect.Field;
8   import java.lang.reflect.InvocationTargetException;
9   import java.lang.reflect.Modifier;
10  import java.util.ArrayList;
11  import java.util.Arrays;
12  import java.util.Collection;
13  import java.util.HashMap;
14  import java.util.Map;
15  import java.util.Optional;
16  import org.apache.maven.plugin.AbstractMojo;
17  import org.apache.maven.plugin.MojoExecutionException;
18  import org.apache.maven.plugin.MojoFailureException;
19  import org.cactoos.iterable.Mapped;
20  import org.cactoos.list.ListOf;
21  import org.cactoos.set.SetOf;
22  
23  /**
24   * Mutable mojo builder.
25   *
26   * @param <T> Type of mojo
27   * @since 0.1
28   */
29  final class Moja<T extends AbstractMojo> {
30  
31      /**
32       * The type of mojo.
33       */
34      private final Class<T> type;
35  
36      /**
37       * All attributes.
38       */
39      private final Map<String, Object> attrs;
40  
41      /**
42       * Ctor.
43       *
44       * @param mojo The type of Mojo
45       */
46      Moja(final Class<T> mojo) {
47          this.type = mojo;
48          this.attrs = new HashMap<>(0);
49      }
50  
51      @Override
52      public String toString() {
53          return String.format("Moja<%s>", this.type.getSimpleName());
54      }
55  
56      /**
57       * Add one more attribute and return self.
58       *
59       * @param attr The name
60       * @param value The value
61       * @return Itself
62       */
63      Moja<T> with(final String attr, final Object value) {
64          this.attrs.put(attr, value);
65          return this;
66      }
67  
68      /**
69       * Copy attributes from the given Mojo.
70       *
71       * @param mojo Another mojo
72       * @return Itself
73       */
74      @SuppressWarnings("PMD.AvoidAccessibilityAlteration")
75      Moja<T> copy(final Object mojo) {
76          final Collection<String> mine = new ListOf<>(
77              new Mapped<>(
78                  Field::getName,
79                  Moja.fields(this.type)
80              )
81          );
82          for (final Field field : new SetOf<>(Moja.fields(mojo.getClass()))) {
83              if (!mine.contains(field.getName())) {
84                  continue;
85              }
86              if (Modifier.isStatic(field.getModifiers())) {
87                  continue;
88              }
89              try {
90                  field.setAccessible(true);
91                  this.with(field.getName(), field.get(mojo));
92              } catch (final IllegalAccessException ex) {
93                  throw new IllegalStateException(ex);
94              }
95          }
96          return this;
97      }
98  
99      /**
100      * Execute it.
101      */
102     void execute() {
103         try {
104             final AbstractMojo mojo = this.type.getConstructor().newInstance();
105             for (final Map.Entry<String, Object> ent : this.attrs.entrySet()) {
106                 this.initField(this.type, mojo, ent);
107             }
108             mojo.execute();
109         } catch (final MojoExecutionException
110             | MojoFailureException
111             | InstantiationException
112             | IllegalAccessException
113             | NoSuchMethodException
114             | InvocationTargetException ex
115         ) {
116             throw new IllegalStateException(ex);
117         }
118     }
119 
120     /**
121      * List all fields of a class.
122      * @param cls The class
123      * @return List of fields
124      */
125     private static Collection<Field> fields(final Class<?> cls) {
126         final Collection<Field> fields = new ArrayList<>(0);
127         Class<?> clazz = cls;
128         while (!clazz.equals(Object.class)) {
129             fields.addAll(Arrays.asList(clazz.getDeclaredFields()));
130             clazz = clazz.getSuperclass();
131         }
132         return fields;
133     }
134 
135     /**
136      * Init a field.
137      * @param clazz The mojo class
138      * @param mojo The mojo
139      * @param entry Field name and value
140      * @throws java.lang.IllegalAccessException If can't set field.
141      */
142     @SuppressWarnings("PMD.AvoidAccessibilityAlteration")
143     private void initField(
144         final Class<?> clazz,
145         final AbstractMojo mojo,
146         final Map.Entry<String, Object> entry
147     ) throws IllegalAccessException {
148         final String name = entry.getKey();
149         final Optional<Field> declared = Arrays.stream(clazz.getDeclaredFields())
150             .filter(f -> f.getName().equals(name))
151             .findFirst();
152         if (declared.isPresent()) {
153             final Field field = declared.get();
154             field.setAccessible(true);
155             field.set(mojo, entry.getValue());
156         } else {
157             final Class<?> parent = clazz.getSuperclass();
158             if (parent == null) {
159                 throw new IllegalStateException(
160                     String.format(
161                         "Can't find '%s' in '%s'",
162                         name,
163                         this.type.getCanonicalName()
164                     )
165                 );
166             } else {
167                 this.initField(parent, mojo, entry);
168             }
169         }
170     }
171 }
