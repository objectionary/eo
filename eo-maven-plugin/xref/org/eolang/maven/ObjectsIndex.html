5   package org.eolang.maven;
6   
7   import com.jcabi.aspects.RetryOnFailure;
8   import java.net.URL;
9   import java.util.Set;
10  import java.util.concurrent.TimeUnit;
11  import org.cactoos.Scalar;
12  import org.cactoos.Text;
13  import org.cactoos.iterable.Mapped;
14  import org.cactoos.scalar.Sticky;
15  import org.cactoos.scalar.Unchecked;
16  import org.cactoos.set.SetOf;
17  import org.cactoos.text.Split;
18  import org.cactoos.text.TextOf;
19  
20  /**
21   * The objects index that contains all available EO objects.
22   *
23   * @see <a href="https://github.com/objectionary/home/blob/gh-pages/objectionary.lst">Example</a>
24   * @since 0.29
25   */
26  final class ObjectsIndex {
27  
28      /**
29       * Tags.
30       */
31      private static final String HOME = "https://home.objectionary.com/objectionary.lst";
32  
33      /**
34       * Cached objects index.
35       */
36      private final Scalar<? extends Set<String>> objects;
37  
38      /**
39       * Ctor.
40       */
41      ObjectsIndex() {
42          this(
43              () -> new SetOf<>(
44                  new Mapped<>(
45                      ObjectsIndex::convert,
46                      new Mapped<>(
47                          Text::asString,
48                          new Split(
49                              ObjectsIndex.asText(new URL(ObjectsIndex.HOME)),
50                              "\n"
51                          )
52                      )
53                  )
54              )
55          );
56      }
57  
58      /**
59       * Ctor.
60       * @param all All objects index.
61       */
62      ObjectsIndex(final Scalar<? extends Set<String>> all) {
63          this.objects = new Sticky<>(all);
64      }
65  
66      /**
67       * Checks whether object index contains the object.
68       *
69       * @param name Object name.
70       * @return True if object index contains the object.
71       * @throws Exception If something unexpected happened.
72       */
73      public boolean contains(final String name) throws Exception {
74          return this.objects.value().contains(name);
75      }
76  
77      /**
78       * Converts object name to the format that is used in the objectionary.
79       * - "objects/org/eolang/array.eo" -> "org.eolang.array"
80       * - "tests/org/eolang/seq-tests.eo" -> "org.eolang.seq-tests"
81       * @param name Object name in raw format.
82       * @return Object name in objectionary format.
83       */
84      private static String convert(final String name) {
85          return name.substring(0, name.length() - 3)
86              .replace('/', '.')
87              .substring(name.indexOf('/') + 1);
88      }
89  
90      /**
91       * Download from the URL and return the content.
92       * @param url The URL with tags
93       * @return The body of the web page
94       */
95      @RetryOnFailure(delay = 1L, unit = TimeUnit.SECONDS)
96      private static Text asText(final URL url) {
97          return new TextOf(
98              new Unchecked<>(() -> new TextOf(url).asString())
99          );
100     }
101 }
