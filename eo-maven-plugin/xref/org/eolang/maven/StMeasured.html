5   package org.eolang.maven;
6   
7   import com.jcabi.xml.XML;
8   import com.yegor256.xsline.Shift;
9   import java.io.IOException;
10  import java.nio.file.Files;
11  import java.nio.file.Path;
12  import java.nio.file.StandardOpenOption;
13  
14  /**
15   * Shift that measures and saves stats into a file.
16   *
17   * @since 0.30
18   */
19  final class StMeasured implements Shift {
20  
21      /**
22       * Origin shift.
23       */
24      private final Shift origin;
25  
26      /**
27       * Log file.
28       */
29      private final Path path;
30  
31      /**
32       * Ctor.
33       * @param shift Origin shift
34       * @param log Log file
35       */
36      StMeasured(final Shift shift, final Path log) {
37          this.origin = shift;
38          this.path = log;
39      }
40  
41      @Override
42      public String uid() {
43          return this.origin.uid();
44      }
45  
46      @Override
47      @SuppressWarnings("PMD.PrematureDeclaration")
48      public XML apply(final int position, final XML xml) {
49          final long start = System.currentTimeMillis();
50          final XML out = this.origin.apply(position, xml);
51          try {
52              Files.write(
53                  this.path,
54                  String.format(
55                      "%s,%d\n",
56                      this.origin.uid(),
57                      System.currentTimeMillis() - start
58                  ).getBytes(),
59                  StandardOpenOption.APPEND,
60                  StandardOpenOption.CREATE
61              );
62          } catch (final IOException ex) {
63              throw new IllegalArgumentException(ex);
64          }
65          return out;
66      }
67  }
