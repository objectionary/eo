5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import org.apache.maven.plugins.annotations.LifecyclePhase;
9   import org.apache.maven.plugins.annotations.Mojo;
10  
11  /**
12   * It deletes binary files, which were previously copied by "place" mojo so
13   * these binaries are not got into result JAR.
14   * @since 0.11
15   */
16  @Mojo(
17      name = "unplace",
18      defaultPhase = LifecyclePhase.PREPARE_PACKAGE,
19      threadSafe = true
20  )
21  public final class MjUnplace extends MjSafe {
22  
23      @Override
24      public void exec() throws IOException {
25          new Timed(
26              new Unplacing(
27                  this.placedTojos,
28                  this.classesDir.toPath(),
29                  this.keepBinaries
30              )
31          ).exec();
32      }
33  }
