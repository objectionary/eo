37  /**
38   * Transpile.
39   * <p>
40   *     This Maven Mojo class transpiles XMIR files into Java source files.
41   *     The goal of this class is to automate the process of transforming EO
42   *     intermediate representations (XMIRs) into Java code, optimize the output,
43   *     and integrate the results into the Maven build lifecycle for further
44   *     processing.
45   * </p>
46   * <p>
47   *     This is one of the final goals that transpiles XMIRs to Java files generated by
48   *     {@link MjAssemble}, {@link MjParse} or {@link MjPull} goals.
49   *     The resulting Java files are stored in the {@link #DIR} directory.
50   *     The intermediate optimized XMIRs are stored in the {@link #PRE} directory.
51   * </p>
52   * @since 0.1
53   * @checkstyle ClassFanOutComplexityCheck (400 lines)
54   */
55  @Mojo(
56      name = "transpile",
57      defaultPhase = LifecyclePhase.PROCESS_SOURCES,
58      threadSafe = true,
59      requiresDependencyResolution = ResolutionScope.COMPILE
60  )
61  public final class MjTranspile extends MjSafe {

124     @Override
125     public void exec() throws IOException {
126         final long begin = System.currentTimeMillis();
127         final Collection<TjForeign> sources = this.scopedTojos().withXmir();
128         final int saved = new Threaded<>(
129             sources,
130             this::transpiled
131         ).total() + new PackageInfos(this.generatedDir.toPath()).create();
132         Logger.info(
133             this, "Transpiled %d XMIRs, created %d Java files in %[file]s",
134             sources.size(), saved, this.generatedDir
135         );
136         if (this.addSourcesRoot) {
137             this.project.addCompileSourceRoot(this.generatedDir.getAbsolutePath());
138             Logger.info(
139                 this, "The directory added to Maven 'compile-source-root': %[file]s",
140                 this.generatedDir
141             );
142             final String gtests = this.generatedDir.toPath().getParent().resolve(
143                 "generated-test-sources"
144             ).toAbsolutePath().toString();
145             this.project.addTestCompileSourceRoot(gtests);
146             Logger.info(
147                 this, "The directory added to Maven 'test-compile-source-root': %[file]s",
148                 gtests
149             );
150         }
151         Logger.info(
152             this,
153             "Transpilation took %[ms]s in total",
154             System.currentTimeMillis() - begin
155         );
156     }

158     /**
159      * Transpile.
160      * @param tojo Tojo that should be transpiled.
161      * @return Number of transpiled files.
162      * @throws java.io.IOException If any issues with I/O
163      */
164     private int transpiled(
165         final TjForeign tojo
166     ) throws IOException {
167         final Path source = tojo.xmir();
168         final XML xmir = new XMLDocument(source);
169         final Path base = this.targetDir.toPath().resolve(MjTranspile.DIR);
170         final Path target = new Place(
171             new OnDetailed(new OnDefault(xmir), source).get()
172         ).make(base, MjAssemble.XMIR);
173         final Supplier<String> hsh = new TojoHash(tojo);
174         final AtomicBoolean rewrite = new AtomicBoolean(false);
175         final Function<XML, XML> transform = this.transpilation(source);
176         final String version = this.plugin.getVersion();
177         final Path tail = base.relativize(target);
178         final Path cdir = this.cache.toPath().resolve(MjTranspile.CACHE);
179         if (this.cacheEnabled) {
180             new ConcurrentCache(
181                 new Cache(
182                     new CachePath(cdir, version, hsh.get()),
183                     src -> {
184                         rewrite.compareAndSet(false, true);
185                         final long start = System.currentTimeMillis();
186                         final String res = transform.apply(xmir).toString();
187                         Logger.debug(
188                             this,
189                             "Transpiled %[file]s (%s) to %[file]s (%s) in %[ms]s (cache miss), version: %s, hash: %s, tail: %s, cache enabled: %b, cache dir: %[file]s",
190                             source,
191                             MjTranspile.info(source),
192                             target,
193                             MjTranspile.info(target),
194                             System.currentTimeMillis() - start,
195                             version,
196                             hsh.get(),
197                             tail,
198                             this.cacheEnabled,
199                             cdir
200                         );
201                         return res;
202                     }
203                 )
204             ).apply(source, target, tail);
205         } else {
206             rewrite.compareAndSet(false, true);
207             new Saved(transform.apply(xmir).toString(), target).value();
208         }
209         return this.javaGenerated(rewrite.get(), target, hsh.get());
210     }

212     /**
213      * File info for logging.
214      * @param info Path to file
215      * @return Info string
216      * @throws IOException If fails
217      */
218     private static String info(final Path info) throws IOException {
219         final String res;
220         if (Files.exists(info)) {
221             res = Files.getLastModifiedTime(info).toString();
222         } else {
223             res = "Not exists yet";
224         }
225         return res;
226     }

228     /**
229      * Transpile XSL transformations.
230      * If {@link MjSafe#trackTransformationSteps} is {@code true} - we create new {@link Xsline}
231      * for every XMIR in purpose of thread safety.
232      * @param source Path to source XMIR
233      * @return XSL transformations that transpiles XMIR to Java.
234      */
235     private Function<XML, XML> transpilation(final Path source) {
236         final Train<Shift> measured = this.measured(MjTranspile.TRAIN);
237         final Function<XML, XML> func;
238         if (this.trackTransformationSteps) {
239             func = xml -> new Xsline(
240                 new TrSpy(
241                     measured,
242                     new StickyFunc<>(
243                         doc -> new Place(
244                             new OnDetailed(new OnDefault(doc), source).get()
245                         ).make(this.targetDir.toPath().resolve(MjTranspile.PRE), "")
246                     )
247                 )
248             ).pass(xml);
249         } else {
250             func = new Xsline(measured)::pass;
251         }
252         return func;
253     }

255     /**
256      * Generate java files and count them.
257      * @param rewrite Rewrite .java files even if they exist
258      * @param target Full target path to XMIR after transpilation optimizations
259      * @param hsh Tojo hash
260      * @return Amount of generated .java files
261      * @throws IOException If fails to save files
262      */
263     private int javaGenerated(
264         final boolean rewrite,
265         final Path target,
266         final String hsh
267     ) throws IOException {
268         final long begin = System.currentTimeMillis();
269         final AtomicInteger saved = new AtomicInteger(0);
270         if (Files.exists(target)) {
271             final Xnav object = new Xnav(target).element("object");
272             final Collection<Xnav> classes = object.elements(Filter.withName("class"))
273                 .collect(Collectors.toList());
274             final boolean atom = object.path("/object/o/o[@name='λ']").findAny().isPresent();
275             for (final Xnav clazz : classes) {
276                 final String jname = clazz.attribute("java-name").text().get();
277                 if (!atom || jname.endsWith("Test")) {
278                     final Path tgt = new Place(jname).make(
279                         this.generatedDir.toPath(), MjTranspile.JAVA
280                     );
281                     final Footprint java = new FpJavaGenerated(
282                         clazz, new FileGenerationReport(saved, tgt, target)
283                     );
284                     new JavaPlaced(
285                         new FpIfReleased(
286                             hsh,
287                             new FpAppliedWithCache(
288                                 java,
289                                 this.cached(hsh, jname),
290                                 new RewritePolicy(rewrite, tgt),
291                                 this.cacheEnabled
292                             ),
293                             java
294                         ),
295                         tgt,
296                         this.generatedDir.toPath()
297                     ).exec(clazz, this.transpileTests);
298                 }
299             }
300             Logger.debug(
301                 this,
302                 "Generated %d Java files from %[file]s in %[ms]s",
303                 saved.get(), target, System.currentTimeMillis() - begin
304             );
305         }
306         return saved.get();
307     }

309     /**
310      * Cached path.
311      * @param hsh Hash
312      * @param jname Java class name
313      * @return Supplier of cached path
314      */
315     private Supplier<Path> cached(final String hsh, final String jname) {
316         return new CachePath(
317             this.cache.toPath().resolve(MjTranspile.CACHE),
318             this.plugin.getVersion(),
319             hsh,
320             this.generatedDir.toPath().relativize(
321                 new Place(jname).make(
322                     this.generatedDir.toPath(), MjTranspile.JAVA
323                 )
324             )
325         );
326     }
