5   package org.eolang.maven;
6   
7   import com.jcabi.log.Logger;
8   import java.nio.file.Path;
9   import java.util.concurrent.atomic.AtomicInteger;
10  
11  /**
12   * File generation report.
13   * @since 0.56.7
14   */
15  final class FileGenerationReport {
16  
17      /**
18       * Saved counter.
19       */
20      private final AtomicInteger counter;
21  
22      /**
23       * Generated directory.
24       */
25      private final Path generated;
26  
27      /**
28       * Target directory.
29       */
30      private final Path target;
31  
32      /**
33       * Ctor.
34       * @param aggregate Saved counter
35       * @param gen Generated directory
36       * @param tgt Target directory
37       */
38      FileGenerationReport(final AtomicInteger aggregate, final Path gen, final Path tgt) {
39          this.counter = aggregate;
40          this.generated = gen;
41          this.target = tgt;
42      }
43  
44      /**
45       * Increment and log the file generation.
46       */
47      void incrementAndLog() {
48          this.counter.incrementAndGet();
49          Logger.debug(
50              this,
51              "Generated %[file]s (%[size]s) file from %[file]s (%[size]s)",
52              this.generated, this.generated.toFile().length(),
53              this.target, this.target.toFile().length()
54          );
55      }
56  }
