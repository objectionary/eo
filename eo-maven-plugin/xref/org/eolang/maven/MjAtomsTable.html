7   import com.github.lombrozo.xnav.Xnav;
8   import com.jcabi.log.Logger;
9   import com.jcabi.xml.XML;
10  import com.jcabi.xml.XMLDocument;
11  import com.yegor256.xsline.StClasspath;
12  import com.yegor256.xsline.Xsline;
13  import java.io.File;
14  import java.io.IOException;
15  import java.nio.charset.StandardCharsets;
16  import java.nio.file.Files;
17  import java.nio.file.Path;
18  import java.util.Map;
19  import java.util.TreeMap;
20  import org.apache.maven.plugins.annotations.LifecyclePhase;
21  import org.apache.maven.plugins.annotations.Mojo;
22  import org.apache.maven.plugins.annotations.Parameter;
23  import org.cactoos.text.TextOf;
24  import org.cactoos.text.UncheckedText;

26  /**
27   * Generate the atom return types table from XMIR sources.
28   * <p>
29   * Walks the XMIR sources in {@link #atomsInputDir}, extracts the
30   * {@code forma} of every lambda atom together with its declared return
31   * type, and writes the result as a CSV file at {@link #atomsOutput}.
32   * Each output line has the form {@code <forma>,<return-type>}; entries
33   * are sorted by {@code forma} to make the file stable across builds.
34   * </p>
35   * @since 0.57
36   */
37  @Mojo(
38      name = "atoms-table",
39      defaultPhase = LifecyclePhase.PROCESS_CLASSES,
40      threadSafe = true
41  )
42  public final class MjAtomsTable extends MjSafe {
43  
44      /**
45       * XSL that extracts atom entries from a single XMIR file.
46       */
47      private static final String XSL = "/org/eolang/maven/atoms-table.xsl";
48  
49      /**
50       * Directory with XMIR sources to scan for atoms.
51       * @checkstyle MemberNameCheck (10 lines)
52       */
53      @Parameter(
54          property = "eo.atomsInputDir",
55          required = true,
56          defaultValue = "${project.build.directory}/eo/1-parse"
57      )
58      private File atomsInputDir;
59  
60      /**
61       * Output CSV file with the atoms table.
62       * @checkstyle MemberNameCheck (10 lines)
63       */
64      @Parameter(
65          property = "eo.atomsOutput",
66          required = true,
67          defaultValue = "${project.build.outputDirectory}/org/eolang/atoms.csv"
68      )
69      private File atomsOutput;
70  
71      @Override
72      void exec() throws IOException {
73          final Path home = this.atomsInputDir.toPath();
74          if (!Files.isDirectory(home)) {
75              Logger.info(
76                  this,
77                  "No XMIR sources at %[file]s, atoms table will be empty",
78                  home
79              );
80              this.write(new TreeMap<>());
81              return;
82          }
83          final Map<String, String> table = new TreeMap<>();
84          final Xsline xsline = new Xsline(new StClasspath(MjAtomsTable.XSL));
85          for (final Path source : new Walk(home)) {
86              final XML before = new XMLDocument(
87                  new UncheckedText(new TextOf(source)).asString()
88              );
89              final Xnav after = new Xnav(xsline.pass(before).inner());
90              after.path("/atoms/atom").forEach(
91                  atom -> table.put(
92                      atom.attribute("forma").text().orElseThrow(
93                          () -> new IllegalStateException(
94                              "atom element must carry 'forma' attribute"
95                          )
96                      ),
97                      atom.attribute("returns").text().orElseThrow(
98                          () -> new IllegalStateException(
99                              "atom element must carry 'returns' attribute"
100                         )
101                     )
102                 )
103             );
104         }
105         this.write(table);
106         Logger.info(
107             this,
108             "Wrote %d atom return type(s) to %[file]s",
109             table.size(),
110             this.atomsOutput.toPath()
111         );
112     }
113 
114     /**
115      * Write the atoms table to the output CSV file.
116      * @param table Forma to return-type mappings, already sorted
117      * @throws IOException If write fails
118      */
119     private void write(final Map<String, String> table) throws IOException {
120         final Path target = this.atomsOutput.toPath();
121         Files.createDirectories(target.getParent());
122         final StringBuilder out = new StringBuilder();
123         for (final Map.Entry<String, String> entry : table.entrySet()) {
124             out.append(entry.getKey())
125                 .append(',')
126                 .append(entry.getValue())
127                 .append('\n');
128         }
129         Files.writeString(target, out.toString(), StandardCharsets.UTF_8);
130     }
131 }
