6   package org.eolang.maven;
7   
8   import com.yegor256.WeAreOnline;
9   import java.util.stream.Collectors;
10  import java.util.stream.Stream;
11  import java.util.stream.StreamSupport;
12  import org.cactoos.experimental.Threads;
13  import org.hamcrest.MatcherAssert;
14  import org.hamcrest.Matchers;
15  import org.junit.jupiter.api.Assertions;
16  import org.junit.jupiter.api.Test;
17  import org.junit.jupiter.api.extension.ExtendWith;
18  
19  /**
20   * Test case for {@link ChRemote}.
21   * @since 0.26
22   */
23  @ExtendWith(WeAreOnline.class)
24  final class ChRemoteTest {
25  
26      @Test
27      void getsCommitHashTag() {
28          final String hash = new ChRemote("0.41.1").value();
29          MatcherAssert.assertThat(
30              "ChRemote should return the correct commit hash, but it doesn't",
31              hash,
32              Matchers.equalTo("9a3dee39597b2e9ac305ca57c296b0fa7e10eb55")
33          );
34      }
35  
36      @Test
37      void getsCommitHashOldTag() {
38          final String hash = new ChRemote("0.40.5").value();
39          MatcherAssert.assertThat(
40              "ChRemote should return the correct commit hash for an old tag, but it doesn't",
41              hash,
42              Matchers.equalTo("ee14a1e30a9e0f8f64404e7a52f40c2a07f88359")
43          );
44      }
45  
46      @Test
47      void throwsCommitHashException() {
48          Assertions.assertThrows(
49              IllegalArgumentException.class,
50              () -> new ChRemote("nonsense").value(),
51              "Throws when tag is not found"
52          );
53      }
54  
55      @Test
56      void isThreadSafe() {
57          final int threads = 200;
58          final String sample = new ChRemote("0.40.5").value();
59          MatcherAssert.assertThat(
60              "You can use this class concurrently",
61              StreamSupport.stream(
62                  new Threads<>(
63                      threads,
64                      Stream.generate(
65                          () -> new ChRemote("0.40.5")
66                      ).limit(threads).collect(Collectors.toList())
67                  ).spliterator(), false
68              ).filter(str -> !sample.equals(str)).collect(Collectors.toList()),
69              Matchers.empty()
70          );
71      }
72  }
