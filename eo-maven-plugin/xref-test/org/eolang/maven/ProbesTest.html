26  /**
27   * Test cases for {@link Probes}.
28   *
29   * @since 0.53
30   */
31  final class ProbesTest {
32  
33      @ParameterizedTest
34      @ClasspathSource(value = "org/eolang/maven/probe-packs/", glob = "**.yaml")
35      void checksProbePacks(final String yaml) throws IOException {
36          final Xtory xtory = new XtSticky(new XtYaml(yaml));
37          final List<String> expected = Optional.ofNullable(
38              (List<String>) xtory.map().get("probes")
39          ).orElse(Collections.emptyList());
40          final Probes actual = new Probes(new EoSyntax(xtory.map().get("eo").toString()).parsed());
41          MatcherAssert.assertThat(
42              "We should find the same number of probes as in the YAML file",
43              actual,
44              Matchers.iterableWithSize(expected.size())
45          );
46          MatcherAssert.assertThat(
47              "Probes should match the ones in the YAML file",
48              actual,
49              Matchers.containsInAnyOrder(expected.toArray(new String[0]))
50          );
51      }
52  
53      @Test
54      void findsProbes() throws IOException {
55          MatcherAssert.assertThat(
56              "We should find 11 objects in the program, but not",
57              new Probes(ProbesTest.xmir()),
58              Matchers.allOf(
59                  Matchers.iterableWithSize(11),
60                  Matchers.hasItems(
61                      "org.eolang.while",
62                      "org",
63                      "org.eolang",
64                      "org.eolang.io",
65                      "org.eolang.io.stdout",
66                      "org.eolang.tt",
67                      "org.eolang.tt.sprintf",
68                      "org.eolang.string",
69                      "org.eolang.bytes",
70                      "org.eolang.number",
71                      "org.eolang.number.plus"
72                  )
73              )
74          );
75      }
76  
77      @Test
78      void findsProbesInFile(@TempDir final Path tmp) throws IOException {
79          final Path xmir = tmp.resolve("simple.xmir");
80          Files.write(
81              xmir,
82              new EoSyntax(
83                  String.join(
84                      "\n",
85                      "# No comments.",
86                      "[] > simple",
87                      "  QQ.io.stdout > @",
88                      "    \"Hello, world!\""
89                  )
90              ).parsed().toString().getBytes(StandardCharsets.UTF_8)
91          );
92          MatcherAssert.assertThat(
93              "We should find 6 objects in simple program, but not",
94              new Probes(xmir),
95              Matchers.allOf(
96                  Matchers.iterableWithSize(6),
97                  Matchers.hasItems(
98                      "org",
99                      "org.eolang",
100                     "org.eolang.io",
101                     "org.eolang.io.stdout",
102                     "org.eolang.string",
103                     "org.eolang.bytes"
104                 )
105             )
106         );
107     }
108 
109     @Test
110     void findsProbesInSimpleProgram() throws IOException {
111         MatcherAssert.assertThat(
112             "We should find 6 objects in the program",
113             new Probes(
114                 new EoSyntax(
115                     String.join(
116                         "\n",
117                         "+alias QQ.io.stdout\n",
118                         "stdout \"Hello, world!\" > simple"
119                     )
120                 ).parsed()
121             ),
122             Matchers.allOf(
123                 Matchers.iterableWithSize(6),
124                 Matchers.hasItems(
125                     "org",
126                     "org.eolang",
127                     "org.eolang.io",
128                     "org.eolang.io.stdout",
129                     "org.eolang.string",
130                     "org.eolang.bytes"
131                 )
132             )
133         );
134     }
135 
136     private static XML xmir() throws IOException {
137         return new EoSyntax(
138             String.join(
139                 "\n",
140                 new String[]{
141                     "+package foo.x",
142                     "+also while\n",
143                     "# No comments.",
144                     "[] > main",
145                     "  QQ.io.stdout > @",
146                     "    QQ.tt.sprintf",
147                     "      \"I am %d years old\"",
148                     "      plus.",
149                     "        1337",
150                     "        228",
151                 }
152             )
153         ).parsed();
154     }
