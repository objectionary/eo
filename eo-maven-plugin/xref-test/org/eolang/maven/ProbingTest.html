5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import java.nio.charset.StandardCharsets;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import org.eolang.parser.EoSyntax;
12  import org.hamcrest.MatcherAssert;
13  import org.hamcrest.Matchers;
14  import org.junit.jupiter.api.Test;
15  import org.junit.jupiter.api.io.TempDir;
16  
17  /**
18   * Test cases for {@link Probing}.
19   * @since 0.67.0
20   */
21  final class ProbingTest {
22  
23      @Test
24      void probesSuccessfully(@TempDir final Path temp) throws IOException {
25          final Path xmir = temp.resolve("test.xmir");
26          Files.write(
27              xmir,
28              new EoSyntax(
29                  new HelloWorld().asString()
30              ).parsed().toString().getBytes(StandardCharsets.UTF_8)
31          );
32          final TjsForeign tojos = new TjsForeign();
33          tojos.add("test").withXmir(xmir);
34          new Probing(tojos, new Objectionary.Fake(), true).exec();
35          MatcherAssert.assertThat(
36              "Probe should have found and registered objects from the objectionary",
37              tojos.size(),
38              Matchers.equalTo(8)
39          );
40      }
41  
42      @Test
43      void skipsWhenOffline(@TempDir final Path temp) throws IOException {
44          final Path xmir = temp.resolve("test.xmir");
45          Files.write(
46              xmir,
47              new EoSyntax(
48                  String.join(
49                      System.lineSeparator(),
50                      "# No comments.",
51                      "[] > test",
52                      "  Q.io.stdout > @",
53                      "    \"Hello!\""
54                  )
55              ).parsed().toString().getBytes(StandardCharsets.UTF_8)
56          );
57          final TjsForeign tojos = new TjsForeign();
58          tojos.add("test").withXmir(xmir);
59          new Probing(tojos, new Objectionary.Fake(), false).exec();
60          MatcherAssert.assertThat(
61              "Probe should not register any objects when offline",
62              tojos.size(),
63              Matchers.equalTo(1)
64          );
65      }
66  }
