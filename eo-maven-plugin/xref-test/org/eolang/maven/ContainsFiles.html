5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import java.nio.file.FileSystems;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.util.Arrays;
12  import org.hamcrest.Description;
13  import org.hamcrest.TypeSafeMatcher;
14  
15  /**
16   * Asserting that path contains a files matching provided globs.
17   * @since 0.31.0
18   */
19  @SuppressWarnings({
20      "JTCOP.RuleAllTestsHaveProductionClass",
21      "JTCOP.RuleCorrectTestName",
22      "JTCOP.RuleInheritanceInTests"
23  })
24  final class ContainsFiles extends TypeSafeMatcher<Path> {
25      /**
26       * Patterns.
27       */
28      private final String[] globs;
29  
30      /**
31       * Ctor.
32       * @param glbs Patterns
33       */
34      ContainsFiles(final String... glbs) {
35          this.globs = Arrays.copyOf(glbs, glbs.length);
36      }
37  
38      @Override
39      public void describeTo(final Description description) {
40          description.appendText(String.format("Matching globs: %s", Arrays.toString(this.globs)));
41      }
42  
43      @Override
44      public boolean matchesSafely(final Path path) {
45          return Arrays.stream(this.globs)
46              .anyMatch(
47                  glob -> ContainsFiles.matchesGlob(
48                      path,
49                      glob
50                  )
51              );
52      }
53  
54      /**
55       * Returns whether a path matches a file pattern.
56       * @param item The path.
57       * @param glob The file pattern.
58       * @return True if the item matches the glob.
59       */
60      private static boolean matchesGlob(final Path item, final String glob) {
61          try {
62              return Files.walk(item)
63                  .anyMatch(
64                      FileSystems
65                          .getDefault()
66                          .getPathMatcher(
67                              String.format(
68                                  "glob:%s",
69                                  glob
70                              )
71                          )::matches
72                  );
73          } catch (final IOException ex) {
74              throw new IllegalStateException(
75                  String.format(
76                      "Error while matching glob=`%s` for %s",
77                      glob,
78                      item
79                  ),
80                  ex
81              );
82          }
83      }
84  }
