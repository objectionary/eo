5   package org.eolang.maven;
6   
7   import org.junit.jupiter.api.extension.ExtensionContext;
8   import org.junit.jupiter.api.extension.ParameterContext;
9   import org.junit.jupiter.api.extension.ParameterResolver;
10  
11  /**
12   * This class is instantiated and then called by JUnit when
13   * an argument of a test method is marked with the {@link RandomProgram}
14   * annotation.
15   *
16   * @since 0.42.0
17   */
18  public final class RandomProgramResolver implements ParameterResolver {
19  
20      @Override
21      public boolean supportsParameter(final ParameterContext context,
22          final ExtensionContext ext) {
23          return context.getParameter().getType().equals(String.class)
24              && context.isAnnotated(RandomProgram.class);
25      }
26  
27      @Override
28      public Object resolveParameter(final ParameterContext context, final ExtensionContext ext) {
29          return String.join(
30              "\n",
31              "# This is a random program in EO, which supposedly",
32              "# complies with all syntactic rules of the language,",
33              "# include the requirements for comments.",
34              "[] > foo",
35              "  QQ.io.stdout > @",
36              "    \"Hello, world!\\n\"",
37              ""
38          );
39      }
40  
41  }
