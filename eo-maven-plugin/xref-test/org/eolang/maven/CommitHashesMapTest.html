5   package org.eolang.maven;
6   
7   import java.util.Map;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.params.ParameterizedTest;
11  import org.junit.jupiter.params.provider.CsvSource;
12  
13  /**
14   * Test case for {@link CommitHashesMap}.
15   *
16   * @since 0.29.5
17   */
18  final class CommitHashesMapTest {
19  
20      /**
21       * Check if commit hashes as map contains given tag as key and hash by tag.
22       * @param tag Tag.
23       * @param hash Hash.
24       * @checkstyle AnnotationUseStyleCheck (40 lines)
25       */
26      @ParameterizedTest
27      @CsvSource({
28          "0.26.0, e0b7836",
29          "0.28.10, 9b88393",
30      })
31      void containsValidHash(final String tag, final String hash) {
32          final Map<String, CommitHash> hashes = new CommitHashesMap.Fake();
33          MatcherAssert.assertThat(
34              String.format(
35                  "Commit hashes should have contained tag %s, but they didn't",
36                  tag
37              ),
38              hashes,
39              Matchers.hasKey(tag)
40          );
41          MatcherAssert.assertThat(
42              String.format(
43                  "Commit hashes should have contained hash %s by tag %s, but they didn't",
44                  hash,
45                  tag
46              ),
47              hashes.get(tag).value(),
48              Matchers.equalTo(hash)
49          );
50      }
51  
52      @ParameterizedTest
53      @CsvSource({
54          "0.26.1",
55          "100.100.100",
56      })
57      void doesNotContainTag(final String tag) {
58          MatcherAssert.assertThat(
59              String.format(
60                  "Commit hashes should have contained tag %s, but they didn't",
61                  tag
62              ),
63              new CommitHashesMap(),
64              Matchers.not(
65                  Matchers.hasKey(tag)
66              )
67          );
68      }
69  }
