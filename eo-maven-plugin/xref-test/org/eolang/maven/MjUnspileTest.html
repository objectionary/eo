5   package org.eolang.maven;
6   
7   import com.yegor256.Mktmp;
8   import com.yegor256.MktmpResolver;
9   import java.io.IOException;
10  import java.nio.file.Files;
11  import java.nio.file.Path;
12  import java.nio.file.Paths;
13  import java.util.Set;
14  import org.hamcrest.MatcherAssert;
15  import org.hamcrest.Matchers;
16  import org.junit.jupiter.api.Test;
17  import org.junit.jupiter.api.extension.ExtendWith;
18  
19  /**
20   * Test case for {@link MjUnspile}.
21   *
22   * @since 0.1
23   */
24  @ExtendWith(MktmpResolver.class)
25  final class MjUnspileTest {
26  
27      @Test
28      void deletesClassIfCompiledFromGeneratedSources(@Mktmp final Path temp) throws IOException {
29          final Path foo = Paths.get("a/b/c/foo.class");
30          final FakeMaven maven = new FakeMaven(temp);
31          new Saved("abc", temp.resolve(foo)).value();
32          new Saved("xxx", maven.generatedPath().resolve("a/b/c/foo.java")).value();
33          maven.execute(MjUnspile.class);
34          MatcherAssert.assertThat(
35              String.format("UnspileMojo unable to remove %s class", foo),
36              Files.exists(foo),
37              Matchers.is(false)
38          );
39      }
40  
41      @Test
42      void keepsSpecifiedClasses(@Mktmp final Path temp) throws IOException {
43          final FakeMaven maven = new FakeMaven(temp)
44              .with("keepBinaries", Set.of("EOorg/package-info.class"));
45          new Saved("gen1", maven.generatedPath().resolve("EOorg/package-info.java")).value();
46          new Saved(
47              "gen2", maven.generatedPath().resolve("EOorg/EOeolang/package-info.java")
48          ).value();
49          final Path org = new Saved(
50              "clz", maven.classesPath().resolve("EOorg/package-info.class")
51          ).value();
52          final Path eolang = new Saved(
53              "pkg", maven.classesPath().resolve("EOorg/EOeolang/package-info.class")
54          ).value();
55          MatcherAssert.assertThat(
56              "UnspileMojo must keep files matching to keepBinaries globs",
57              maven.execute(MjUnspile.class).result(),
58              Matchers.allOf(
59                  Matchers.hasValue(org),
60                  Matchers.not(Matchers.hasValue(eolang))
61              )
62          );
63      }
64  
65      @Test
66      void deletesInnerGeneratedClasses(@Mktmp final Path temp) throws IOException {
67          final FakeMaven maven = new FakeMaven(temp);
68          new Saved("outer", maven.generatedPath().resolve("EOorg/EOnumber.java")).value();
69          final Path clazz = new Saved(
70              "clz", maven.classesPath().resolve("EOorg/EOnumber.class")
71          ).value();
72          final Path inner = new Saved(
73              "inner", maven.classesPath().resolve("EOorg/EOnumber$1$2$3.class")
74          ).value();
75          final Path located = new Saved(
76              "clss", maven.classesPath().resolve("EOorg/EOnumber$EOΦorgeolanginner.class")
77          ).value();
78          MatcherAssert.assertThat(
79              "UnspileMojo must delete inner auto generated classes",
80              maven.execute(MjUnspile.class).result(),
81              Matchers.allOf(
82                  Matchers.not(Matchers.hasValue(clazz)),
83                  Matchers.not(Matchers.hasValue(inner)),
84                  Matchers.not(Matchers.hasValue(located))
85              )
86          );
87      }
88  
89      @Test
90      void deletesEmptyDirectories(@Mktmp final Path temp) throws IOException {
91          final FakeMaven maven = new FakeMaven(temp);
92          new Saved("src", maven.generatedPath().resolve("org/eolang/number.java")).value();
93          new Saved("cnt", maven.classesPath().resolve("org/eolang/number.class")).value();
94          MatcherAssert.assertThat(
95              "UnspileMojo must remove all empty directories recursively",
96              maven.execute(MjUnspile.class).result(),
97              Matchers.allOf(
98                  Matchers.not(Matchers.hasKey("target/classes/org/eolang")),
99                  Matchers.not(Matchers.hasKey("target/classes/org")),
100                 Matchers.hasKey("target/classes")
101             )
102         );
103     }
