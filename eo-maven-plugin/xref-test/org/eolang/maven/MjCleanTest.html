9   import com.yegor256.WeAreOnline;
10  import java.io.IOException;
11  import java.nio.file.Files;
12  import java.nio.file.Path;
13  import org.cactoos.set.SetOf;
14  import org.hamcrest.MatcherAssert;
15  import org.hamcrest.Matchers;
16  import org.junit.jupiter.api.Test;
17  import org.junit.jupiter.api.extension.ExtendWith;
18  
19  /**
20   * Test case for {@link MjClean}.
21   *
22   * @since 0.28.6
23   */
24  @ExtendWith(MktmpResolver.class)
25  final class MjCleanTest {
26  
27      @Test
28      void cleansSuccessfully(@Mktmp final Path temp) throws IOException {
29          final Path dir = Files.createDirectories(temp.resolve("target"));
30          final Path out = Files.createDirectories(dir.resolve("child"));
31          final Path small = Files.createDirectories(out.resolve("child.eo"));
32          final Path file = Files.createTempFile(dir, "some", ".eo");
33          if (!small.toFile().exists() || !file.toFile().exists()) {
34              throw new IllegalStateException("Files not created.");
35          }
36          new FakeMaven(temp)
37              .with("targetDir", dir.toFile())
38              .execute(MjClean.class);
39          MatcherAssert.assertThat(
40              "CleanMojo should delete all temp files and directories, but it doesn't",
41              !file.toFile().exists() && !small.toFile().exists(),
42              Matchers.is(true)
43          );
44      }
45  
46      @Test
47      @ExtendWith(WeAreOnline.class)
48      void makesFullCompilingLifecycleSuccessfully(@Mktmp final Path temp) throws IOException {
49          new FakeMaven(temp)
50              .withHelloWorld()
51              .with("includeSources", new SetOf<>("**.eo"))
52              .with("classesDir", temp.resolve("out").toFile())
53              .with("placed", temp.resolve("list").toFile())
54              .with("cache", temp.resolve("cache/parsed").toFile())
55              .with("skipZeroVersions", true)
56              .with("central", Central.EMPTY)
57              .with("ignoreTransitive", true)
58              .execute(MjRegister.class)
59              .execute(MjAssemble.class)
60              .execute(MjClean.class);
61          MatcherAssert.assertThat(
62              "CleanMojo should delete all files after full compiling lifecycle, but it doesn't",
63              temp.resolve("target").toFile().exists(),
64              Matchers.is(false)
65          );
66      }
67  }
