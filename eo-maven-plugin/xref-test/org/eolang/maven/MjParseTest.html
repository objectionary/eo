38      @Test
39      void parsesSuccessfully(@Mktmp final Path temp) throws Exception {

43              MjParse.DIR,
44              MjAssemble.XMIR

72      @Test
73      @ExtendWith(WeAreOnline.class)
74      void parsesWithCache(@Mktmp final Path temp) throws Exception {
75          final Path cache = temp.resolve("cache");
76          final FakeMaven maven = new FakeMaven(temp)
77              .withProgram("invalid content")
78              .with("cache", cache.toFile());
79          final String expected = new UncheckedText(
80              new TextOf(new ResourceOf("org/eolang/maven/main.xmir"))
81          ).asString();
82          final CommitHash hash = new ChCached(new ChNarrow(new ChRemote("0.40.5")));
83          final Path base = maven.targetPath().resolve(MjParse.DIR);
84          final Path target = new Place("foo.x.main").make(base, MjAssemble.XMIR);
85          final Path tail = base.relativize(target);
86          new Cache(
87              cache.resolve(MjParse.CACHE)
88                  .resolve(FakeMaven.pluginVersion())
89                  .resolve(hash.value()),
90              src -> expected
91          ).apply(maven.programTojo().source(), target, tail);
92          target.toFile().delete();
93          final String actual = String.format(
94              "target/%s/foo/x/main.%s",
95              MjParse.DIR,
96              MjAssemble.XMIR
97          );
98          MatcherAssert.assertThat(
99              String.format("We expect that that %s is taken from the cache, but it didn't", actual),
100             new TextOf(
101                 maven.allTojosWithHash(hash)
102                     .execute(new FakeMaven.Parse())
103                     .result()
104                     .get(actual)
105             ).toString(),
106             Matchers.equalTo(expected)
107         );
108     }

170     /**
171      * The test with high number of eo programs reveals concurrency problems of the ParseMojo.
172      * Since other tests works only with single program - it's hard to find concurrency mistakes.
173      * @param temp Test directory.
174      * @throws IOException If problem with filesystem happened.
175      */
176     @Test
177     void parsesConcurrentlyWithLotsOfPrograms(@Mktmp final Path temp) throws IOException {
178         final FakeMaven maven = new FakeMaven(temp);
179         final int total = 50;
180         for (int program = 0; program < total; ++program) {
181             maven.withProgram(
182                 String.format("+package foo.x\n\n# Program\n[] > main%s", FakeMaven.suffix(program))
183             );
184         }
185         final Map<String, Path> res = maven.execute(new FakeMaven.Parse()).result();
186         for (int program = 0; program < total; ++program) {
187             MatcherAssert.assertThat(
188                 "We have to parse concurrently, but we didn't",
189                 res,
190                 Matchers.hasKey(
191                     String.format(
192                         "target/%s/foo/x/main%s.%s",
193                         MjParse.DIR,
194                         FakeMaven.suffix(program),
195                         MjAssemble.XMIR
196                     )
197                 )
198             );
199         }
200     }

242     @Test
243     void parsesWithTargetCache(@Mktmp final Path temp) throws IOException {
244         final FakeMaven maven = new FakeMaven(temp);
245         final File parsed = maven
246             .withHelloWorld()
247             .execute(new FakeMaven.Parse())
248             .result().get(String.format("target/%s/foo/x/main.%s", MjParse.DIR, MjAssemble.XMIR))
249             .toFile();
250         Files.setLastModifiedTime(
251             parsed.toPath(), FileTime.fromMillis(System.currentTimeMillis() + 60_000L)
252         );
253         final long before = parsed.lastModified();
254         maven.execute(new FakeMaven.Parse());
255         final long after = parsed.lastModified();
256         maven.withProgram(
257             String.join(
258                 "\n",
259                 "[] > foo",
260                 "  boom > @"
261             ),
262             "foo",
263             "foo/x/foo.eo"
264         ).execute(new FakeMaven.Parse());
265         MatcherAssert.assertThat(
266             "Parser re-parsed sources, but it should not",
267             parsed.lastModified(),
268             Matchers.allOf(
269                 Matchers.equalTo(before),
270                 Matchers.equalTo(after)
271             )
272         );

275     /**
276      * The mojo that does nothing, but executes infinitely.
277      * @since 0.29
278      */
279     @Mojo(name = "infinite", defaultPhase = LifecyclePhase.VALIDATE)
280     private static final class Infinite extends MjSafe {
281         @Override
282         public void exec() {
283             try {
284                 Thread.sleep(Long.MAX_VALUE);
285             } catch (final InterruptedException ex) {
286                 Thread.currentThread().interrupt();
287                 throw new IllegalStateException(ex);
288             }
289         }
290     }
