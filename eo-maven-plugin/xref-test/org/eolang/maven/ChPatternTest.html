5   package org.eolang.maven;
6   
7   import org.hamcrest.MatcherAssert;
8   import org.hamcrest.Matchers;
9   import org.junit.jupiter.params.ParameterizedTest;
10  import org.junit.jupiter.params.provider.CsvSource;
11  
12  /**
13   * Test case for {@link ChPattern}.
14   *
15   * @since 0.28.11
16   */
17  class ChPatternTest {
18  
19      /**
20       * Get hash by tag using pattern test.
21       *
22       * @param pattern Pattern
23       * @param tag Tag
24       * @param expected Expected Hash
25       */
26      @ParameterizedTest
27      @CsvSource({
28          "'0.*.*:abc2sd3', '0.0.0', abc2sd3",
29          "'*.*.*:abc2sd3', '2.2.2', abc2sd3",
30          "'0.*.*:abc2sd3', '1.0.0', ''",
31          "'0.*.*:m23ss3h', '0.1.2', 'm23ss3h'",
32          "'0.*.*:m23ss3h,1.*.*:abc2sd3', '0.1.2', 'm23ss3h'",
33          "'0.*.*:m23ss3h, 1.*.*:abc2sd3', '0.1.2', 'm23ss3h'",
34          "'0.*.*:m23ss3h,1.*.*:abc2sd3', '1.1.2', 'abc2sd3'",
35          "'0.*.*:m23ss3h,3.*.*:abc2sd3', '2.1.2', ''",
36          "'3.*.*:m23ss3h,3.1.*:abc2sd3', '3.1.2', 'abc2sd3'",
37          "'3.1.2:m23ss3h,3.1.*:abc2sd3', '3.1.2', 'm23ss3h'",
38          "'master:m23ss3h,3.1.*:abc2sd3', 'master', 'm23ss3h'",
39          "'master:m23ss3h,composite-tag:abc2sd3', 'composite-tag', 'abc2sd3'"
40      })
41      void returnsCorrectHashByPattern(
42          final String pattern,
43          final String tag,
44          final String expected
45      ) {
46          MatcherAssert.assertThat(
47              "ChPattern should return the correct hash, but it doesn't",
48              new ChPattern(pattern, tag).value(),
49              Matchers.equalTo(expected)
50          );
51      }
52  }
