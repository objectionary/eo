5   package org.eolang.maven;
6   
7   import com.yegor256.Mktmp;
8   import com.yegor256.MktmpResolver;
9   import java.nio.file.Path;
10  import org.cactoos.list.ListOf;
11  import org.hamcrest.MatcherAssert;
12  import org.hamcrest.Matchers;
13  import org.junit.jupiter.api.Test;
14  import org.junit.jupiter.api.extension.ExtendWith;
15  
16  /**
17   * Test case for {@link Walk}.
18   *
19   * @since 0.11
20   */
21  @ExtendWith(MktmpResolver.class)
22  final class WalkTest {
23  
24      @Test
25      void findsFilesMatchingGlobPattern(@Mktmp final Path temp) throws Exception {
26          new Saved("", temp.resolve("foo/hello/0.1/EObar/x.bin")).value();
27          new Saved("", temp.resolve("EOxxx/bar")).value();
28          final String pattern = "EO**/*";
29          final int count = 1;
30          MatcherAssert.assertThat(
31              String.format(
32                  "Expected %d file(s) matching pattern '%s'",
33                  count,
34                  pattern
35              ),
36              new Walk(temp).includes(new ListOf<>(pattern)),
37              Matchers.iterableWithSize(count)
38          );
39      }
40  }
