31      @Test
32      void compilesSourceAndAddsToCache(@Mktmp final Path temp) throws Exception {
33          final var base = temp.resolve("cache");
34          Files.createDirectories(base);
35          final var source = temp.resolve("source.eo");
36          Files.writeString(source, "[] > main\n  (stdout \"Hello, EO!\") > @\n");
37          final var target = temp.resolve("target.xmir");
38          final var tail = source.getFileName();
39          final String content = "compiled";
40          new Cache(base, p -> content).apply(source, target, tail);
41          MatcherAssert.assertThat(
42              "Target file must be created from source",
43              Files.readString(target),
44              Matchers.equalTo(content)
45          );
46          MatcherAssert.assertThat(
47              "Cache file must be created",
48              Files.exists(base.resolve(tail)),
49              Matchers.is(true)
50          );
51          MatcherAssert.assertThat(
52              "Hash file must be created",
53              Files.exists(base.resolve(String.format("%s.sha256", tail))),
54              Matchers.is(true)
55          );
56      }

58      @Test
59      void readsFromCacheWhenUnchanged(@Mktmp final Path temp) throws Exception {
60          final var base = temp.resolve("cache-directory");
61          Files.createDirectories(base);
62          final var source = temp.resolve("stdin.eo");
63          Files.writeString(source, "[] > main\n  (stdout \"Hello, EO!\") > @\n");
64          final var target = temp.resolve("stdin.xmir");
65          final var counter = new AtomicInteger(0);
66          final var cache = new Cache(
67              base,
68              p -> String.format("stdin %d", counter.incrementAndGet())
69          );
70          final Path tail = source.getFileName();
71          cache.apply(source, target, tail);
72          MatcherAssert.assertThat(
73              "Compilation should happen only once",
74              counter.get(),
75              Matchers.equalTo(1)
76          );
77          cache.apply(source, target, tail);
78          MatcherAssert.assertThat(
79              "Compilation should not happen again",
80              counter.get(),
81              Matchers.equalTo(1)
82          );
83      }
84  
85      @Test
86      void compilesAgainWhenChanged(@Mktmp final Path temp) throws Exception {
87          final var base = temp.resolve("cache-base-dir");
88          Files.createDirectories(base);
89          final var source = temp.resolve("stdout.eo");
90          Files.writeString(source, "[] > main\n  (stdout \"Hello, EO!\") > @\n");
91          final var target = temp.resolve("stdout.xmir");
92          final var counter = new AtomicInteger(0);
93          final var cache = new Cache(
94              base,
95              p -> String.format("compiled %d", counter.incrementAndGet())
96          );
97          cache.apply(source, target, source.getFileName());
98          MatcherAssert.assertThat(
99              "Compilation should happen once",
100             counter.get(),
101             Matchers.equalTo(1)
102         );
103         Files.writeString(source, "[] > main\n  (stdout \"Hello, EO! Modified\") > @\n");
104         cache.apply(source, target, source.getFileName());
105         MatcherAssert.assertThat(
106             "Compilation should happen again after source change",
107             counter.get(),
108             Matchers.equalTo(2)
109         );
110     }

140     @Test
141     void writesCorrectShaHash(@Mktmp final Path temp) throws IOException, NoSuchAlgorithmException {
142         final var base = temp.resolve("cache");
143         Files.createDirectories(base);
144         final var source = temp.resolve("message.txt");
145         final String msg = "hello";
146         final Charset encoding = StandardCharsets.UTF_8;
147         Files.writeString(source, msg, encoding);
148         final var target = temp.resolve("out.txt");
149         final var tail = source.getFileName();
150         final var cache = new Cache(base, p -> "compiled");
151         cache.apply(source, target, tail);
152         MatcherAssert.assertThat(
153             "SHA-256 hash file has incorrect content",
154             Files.readString(base.resolve(String.format("%s.sha256", tail)), encoding),
155             Matchers.equalTo(
156                 Base64.getEncoder().encodeToString(
157                     MessageDigest.getInstance("SHA-256").digest(msg.getBytes(encoding))
158                 )
159             )
160         );
161     }

163     @Test
164     void generatesCorrectHashForLargeFile(
165         @Mktmp final Path temp
166     ) throws IOException, NoSuchAlgorithmException {
167         final var cache = temp.resolve("cache");
168         Files.createDirectories(cache);
169         final var source = temp.resolve("large.txt");
170         final int lines = 100_000;
171         final StringBuilder builder = new StringBuilder(lines * 10);
172         IntStream.range(0, lines).forEach(
173             i -> builder.append("Line ").append(i).append('\n')
174         );
175         final String content = builder.toString();
176         Files.writeString(source, content, StandardCharsets.UTF_8);
177         final var target = temp.resolve("out.txt");
178         final var tail = source.getFileName();
179         new Cache(cache, p -> content).apply(source, target, tail);
180         MatcherAssert.assertThat(
181             "SHA-256 hash file has incorrect content for large file",
182             Files.readString(
183                 cache.resolve(String.format("%s.sha256", tail)),
184                 StandardCharsets.UTF_8
185             ),
186             Matchers.equalTo(
187                 Base64.getEncoder().encodeToString(
188                     MessageDigest.getInstance("SHA-256")
189                         .digest(content.getBytes(StandardCharsets.UTF_8))
190                 )
191             )
192         );
193     }
