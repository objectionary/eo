135      * You can use utility classes to run predefined maven pipelines:
136      *  - {@link org.eolang.maven.FakeMaven.Parse} to parse eo code
137      *  - see other inner classes below.
138      * @param mojo Several mojos to execute.
139      * @return Workspace after executing Mojo.
140      * @throws IOException If some problem with filesystem is happened.
141      */
142     FakeMaven execute(final Iterable<Class<? extends AbstractMojo>> mojo)
143         throws IOException {
144         for (final Class<? extends AbstractMojo> clazz : mojo) {

161     /**
162      * Executes Mojo in the workspace.
163      *
164      * @param mojo Mojo to execute.
165      * @param <T> Template for descendants of Mojo.
166      * @return Workspace after executing Mojo.
167      * @throws java.io.IOException If some problem with filesystem has happened.
168      * @checkstyle ExecutableStatementCountCheck (100 lines)
169      * @checkstyle JavaNCSSCheck (100 lines)
170      */
171     <T extends AbstractMojo> FakeMaven execute(final Class<T> mojo) throws IOException {
172         if (this.defaults) {
173             final Path placed = Paths.get("placed.json");
174             this.params.putIfAbsent("targetDir", this.targetPath().toFile());
175             this.params.putIfAbsent(
176                 "xslMeasures", this.targetPath().resolve("measures.csv").toFile()
177             );
178             this.params.putIfAbsent("foreign", this.foreignPath().toFile());
179             this.params.putIfAbsent("foreignFormat", "csv");
180             this.params.putIfAbsent("project", new MavenProjectStub());
181             this.params.putIfAbsent("transpiledFormat", "csv");
182             this.params.putIfAbsent("skipZeroVersions", true);
183             this.params.putIfAbsent("cacheEnabled", true);
184             this.params.putIfAbsent("discoverSelf", false);
185             this.params.putIfAbsent("ignoreVersionConflicts", false);
186             this.params.putIfAbsent("ignoreTransitive", true);
187             this.params.putIfAbsent("central", new DummyCentral());
188             this.params.putIfAbsent("resolveInCentral", false);
189             this.params.putIfAbsent("placed", this.workspace.resolve(placed).toFile());
190             this.params.putIfAbsent("placedFormat", "json");
191             this.params.putIfAbsent(
192                 "sourcesDir", this.workspace.resolve(".").toFile()
193             );
194             this.params.putIfAbsent(
195                 "cache", this.workspace.resolve("eo/cache/parsed").toFile()
196             );
197             this.params.putIfAbsent("generatedDir", this.generatedPath().toFile());
198             this.params.putIfAbsent("placedFormat", "csv");
199             this.params.putIfAbsent("plugin", FakeMaven.pluginDescriptor());
200             this.params.putIfAbsent(
201                 "objectionary",
202                 new Synced<>(new ScalarOf<>(Objectionary.Fake::new))
203             );
204             this.params.putIfAbsent("rewriteBinaries", true);
205             this.params.putIfAbsent("offline", false);
206             this.params.putIfAbsent("classesDir", this.classesPath().toFile());
207         }
208         final Moja<T> moja = new Moja<>(mojo);
209         for (final Map.Entry<String, ?> entry : this.allowedParams(mojo).entrySet()) {
210             moja.with(entry.getKey(), entry.getValue());
211         }
212         moja.execute();
213         return this;
214     }

390     /**
391      * Creates of the result map with all files and folders that was created
392      * or compiled during mojo execution.
393      *
394      * @return Map of "relative UNIX path" (key) - "absolute path" (value).
395      * @throws IOException If some problem with filesystem have happened.
396      */
397     Map<String, Path> result() throws IOException {
398         final Path root = this.workspace.resolve("");
399         return Files.walk(root).collect(
400             Collectors.toMap(
401                 p -> String.join(
402                     "/",
403                     root.relativize(p).toString().split(Pattern.quote(File.separator))
404                 ),
405                 Function.identity()
406             )
407         );
408     }

434     static String suffix(final int index) {
435         final String suffix;

437             suffix = "";

439             suffix = String.format("-%d", index);

444     /**
445      * Ensures the map of allowed params for the Mojo.
446      *
447      * @param mojo Mojo
448      * @return Map of params that applicable to the Mojo
449      */
450     private Map<String, ?> allowedParams(final Class<? extends AbstractMojo> mojo) {
451         final Map<String, Object> res = new HashMap<>();
452         final Set<String> allowed = FakeMaven.mojoFields(mojo, new HashSet<>());
453         for (final Map.Entry<String, Object> entry : this.params.entrySet()) {
454             if (allowed.contains(entry.getKey())) {
455                 res.put(entry.getKey(), entry.getValue());
456             }
457         }
458         return res;
459     }

490     /**
491      * Looks for all declared fields for mojo and its parents.
492      *
493      * @param mojo Mojo or mojo parent.
494      * @param fields Already collected fields.
495      * @return All mojo and mojo parent fields.
496      */
497     private static Set<String> mojoFields(final Class<?> mojo, final Set<String> fields) {
498         final Set<String> res;
499         if (mojo == null) {
500             res = fields;
501         } else {
502             Stream.of(mojo.getDeclaredFields()).map(Field::getName).forEach(fields::add);
503             res = mojoFields(mojo.getSuperclass(), fields);
504         }
505         return res;
506     }

648     /**
649      * The class for emulating of Maven Central repository.
650      * DummyCentral creates an empty dependency jar file under the path.
651      *
652      * @since 0.28.11
653      */
654     private static final class DummyCentral implements BiConsumer<Dependency, Path> {
655 
656         @Override
657         public void accept(final Dependency dependency, final Path path) {
658             try {
659                 Files.createDirectories(path);
660                 final String other = DummyCentral.className(dependency);
661                 Files.createFile(path.resolve(other));
662             } catch (final IOException ex) {
663                 throw new IllegalStateException(
664                     String.format("Can't save '%s' to '%s'", dependency, path),
665                     ex
666                 );
667             }
668         }
669 
670         /**
671          * Dependency class name.
672          *
673          * @param dependency Dependency
674          * @return Class file name
675          */
676         private static String className(final Dependency dependency) {
677             final List<String> parts = new ArrayList<>(3);
678             if (dependency.getArtifactId() != null && !dependency.getArtifactId().isEmpty()) {
679                 parts.add(dependency.getArtifactId());
680             }
681             if (dependency.getVersion() != null && !dependency.getVersion().isEmpty()) {
682                 parts.add(dependency.getVersion());
683             }
684             if (dependency.getClassifier() != null && !dependency.getClassifier().isEmpty()) {
685                 parts.add(dependency.getClassifier());
686             }
687             return String.format("%s.class", String.join("-", parts));
688         }
