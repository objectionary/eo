5   package org.eolang.maven;
6   
7   import org.hamcrest.MatcherAssert;
8   import org.hamcrest.Matchers;
9   import org.junit.jupiter.api.Assertions;
10  import org.junit.jupiter.api.Test;
11  import org.junit.jupiter.params.ParameterizedTest;
12  import org.junit.jupiter.params.provider.CsvSource;
13  
14  /**
15   * Test case for {@link ChNarrow}.
16   *
17   * @since 0.28.11
18   */
19  final class ChNarrowTest {
20  
21      @ParameterizedTest
22      @CsvSource({
23          "1234567, 1234567",
24          "12345678, 1234567",
25          "123456789, 1234567",
26          "1, 1"
27      })
28      void cutsHashCorrectly(final String input, final String output) {
29          MatcherAssert.assertThat(
30              String.format(
31                  "The hash of %s was calculated incorrectly",
32                  input
33              ),
34              new ChNarrow(
35                  new CommitHash.ChConstant(input)
36              ).value(),
37              Matchers.equalTo(output)
38          );
39      }
40  
41      @Test
42      void throwsExceptionIfEmpty() {
43          Assertions.assertThrows(
44              IllegalArgumentException.class,
45              () -> new ChNarrow(new CommitHash.ChConstant("")).value(),
46              "An exception should be thrown for an empty commit hash, but it was not"
47          );
48      }
49  
50      @Test
51      void throwsExceptionIfNull() {
52          Assertions.assertThrows(
53              NullPointerException.class,
54              () -> new ChNarrow(null).value(),
55              "An exception should be thrown for a null commit hash, but it was not"
56          );
57      }
58  }
