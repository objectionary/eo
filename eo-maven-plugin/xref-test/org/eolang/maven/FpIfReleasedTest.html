5   package org.eolang.maven;
6   
7   import java.io.IOException;
8   import java.nio.file.Path;
9   import org.junit.jupiter.api.Assertions;
10  import org.junit.jupiter.api.Test;
11  import org.junit.jupiter.api.io.TempDir;
12  
13  /**
14   * Test case for {@link FpIfReleased}.
15   * @since 0.57
16   */
17  final class FpIfReleasedTest {
18  
19      @Test
20      void throwsNpeIfHashIsNull(@TempDir final Path tmp) {
21          Assertions.assertThrows(
22              NullPointerException.class,
23              () -> new FpIfReleased(
24                  () -> null,
25                  (source, target) -> source,
26                  (source, target) -> target
27              ).apply(tmp.resolve("src"), tmp.resolve("tgt")),
28              "Should throw NPE if hash is null"
29          );
30      }
31  
32      @Test
33      void choosesTheFirstFootprintBecauseCacheable(@TempDir final Path tmp) throws IOException {
34          final Path src = tmp.resolve("first");
35          final Path result = new FpIfReleased(
36              () -> "somehash",
37              (source, target) -> source,
38              (source, target) -> target
39          ).apply(src, tmp.resolve("second"));
40          Assertions.assertEquals(
41              src,
42              result,
43              "Should choose the first footprint when cacheable"
44          );
45      }
46  
47      @Test
48      void choosesTheSecondFootprintBecauseHashIsEmpty(@TempDir final Path tmp) throws IOException {
49          final Path tgt = tmp.resolve("right-empty-hash");
50          final Path result = new FpIfReleased(
51              () -> "",
52              (source, target) -> source,
53              (source, target) -> target
54          ).apply(tmp.resolve("left-empty-hash"), tgt);
55          Assertions.assertEquals(
56              tgt,
57              result,
58              "Should choose the second footprint when hash is empty"
59          );
60      }
61  }
