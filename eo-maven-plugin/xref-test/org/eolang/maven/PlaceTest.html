5   package org.eolang.maven;
6   
7   import java.nio.file.Paths;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.api.Test;
11  
12  /**
13   * Test case for {@link Place}.
14   *
15   * @since 0.11
16   */
17  final class PlaceTest {
18  
19      @Test
20      void makesPath() {
21          MatcherAssert.assertThat(
22              "Place must make the path by full name of the object",
23              new Place("hello.foo.bar")
24                  .make(Paths.get("/tmp/test"), MjAssemble.XMIR)
25                  .toString()
26                  .replace("\\", "/"),
27              Matchers.equalTo(String.format("/tmp/test/hello/foo/bar.%s", MjAssemble.XMIR))
28          );
29      }
30  
31      @Test
32      void makesSimplePath() {
33          MatcherAssert.assertThat(
34              "Place must make the simple path",
35              new Place("hey")
36                  .make(Paths.get("/tmp"), "xml")
37                  .toString()
38                  .replace("\\", "/"),
39              Matchers.equalTo("/tmp/hey.xml")
40          );
41      }
42  }
