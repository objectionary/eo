5   package org.eolang.maven;
6   
7   import com.yegor256.Mktmp;
8   import com.yegor256.MktmpResolver;
9   import java.io.IOException;
10  import java.nio.file.Path;
11  import org.cactoos.io.ResourceOf;
12  import org.cactoos.scalar.LengthOf;
13  import org.hamcrest.MatcherAssert;
14  import org.hamcrest.Matchers;
15  import org.junit.jupiter.api.extension.ExtendWith;
16  import org.junit.jupiter.params.ParameterizedTest;
17  import org.junit.jupiter.params.provider.CsvSource;
18  
19  /**
20   * Test case for {@link DpsDepgraph.DpsJson}.
21   *
22   * @since 0.28.11
23   */
24  @ExtendWith(MktmpResolver.class)
25  final class DpsDepgraphTest {
26  
27      @ParameterizedTest
28      @CsvSource({
29          "eo-math-dependencies-transient-dependency.json, 3",
30          "eo-math-dependencies-without-foreign.json, 7"
31      })
32      void readsAllDependenciesFromJsonFile(
33          final String name,
34          final long number,
35          @Mktmp final Path tmp
36      ) throws Exception {
37          MatcherAssert.assertThat(
38              "The number of dependencies must be correct, but it isn't",
39              new LengthOf(new DpsDepgraph.DpsJson(this.file(tmp, name))).value(),
40              Matchers.equalTo(number)
41          );
42      }
43  
44      private Path file(final Path tmp, final String name) {
45          try {
46              return new Saved(
47                  new ResourceOf(
48                      String.format("org/eolang/maven/dependencies/%s", name)
49                  ),
50                  tmp.resolve(name)
51              ).value();
52          } catch (final IOException ex) {
53              throw new IllegalStateException(ex);
54          }
55      }
56  }
