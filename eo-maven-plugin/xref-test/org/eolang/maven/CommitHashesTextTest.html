5   package org.eolang.maven;
6   
7   import com.yegor256.Together;
8   import com.yegor256.WeAreOnline;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.Matchers;
11  import org.junit.jupiter.api.Test;
12  import org.junit.jupiter.api.extension.ExtendWith;
13  
14  /**
15   * Test case for {@link CommitHashesText}.
16   *
17   * @since 0.37.0
18   */
19  @ExtendWith(WeAreOnline.class)
20  final class CommitHashesTextTest {
21  
22      @Test
23      void downloadsDefaultList() throws Exception {
24          MatcherAssert.assertThat(
25              "CommitHashesText downloads the default list of hashes from Objectionary",
26              new CommitHashesText().asString(),
27              Matchers.containsString("master")
28          );
29      }
30  
31      @Test
32      void isThreadSafe() {
33          final CommitHashesText text = new CommitHashesText();
34          MatcherAssert.assertThat(
35              "Can be used in different threads without NPE",
36              new Together<>(
37                  thread -> text.asString() != null
38              ),
39              Matchers.not(Matchers.hasItems(false))
40          );
41      }
42  }
