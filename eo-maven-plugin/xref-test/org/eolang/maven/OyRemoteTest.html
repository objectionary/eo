5   package org.eolang.maven;
6   
7   import com.yegor256.WeAreOnline;
8   import java.io.IOException;
9   import java.net.MalformedURLException;
10  import java.net.URL;
11  import org.hamcrest.MatcherAssert;
12  import org.hamcrest.Matchers;
13  import org.junit.jupiter.api.Assertions;
14  import org.junit.jupiter.api.Test;
15  import org.junit.jupiter.api.extension.ExtendWith;
16  
17  /**
18   * Test for {@link OyRemote}.
19   *
20   * @since 0.1.0
21   */
22  final class OyRemoteTest {
23  
24      @Test
25      void buildsCorrectUrlForProgram() throws Exception {
26          MatcherAssert.assertThat(
27              "OyRemote.UrlOy generates correct URL for program",
28              new OyRemote.UrlOy(
29                  "https://raw/objectionary/home/%s/objects/%s.eo",
30                  "abcde"
31              ).value("org.eolang.app"),
32              Matchers.is(
33                  new URL("https://raw/objectionary/home/abcde/objects/org/eolang/app.eo")
34              )
35          );
36      }
37  
38      @Test
39      void buildsCorrectUrlForDirectory() throws Exception {
40          MatcherAssert.assertThat(
41              "OyRemote.UrlOy generates correct URL for directory",
42              new OyRemote.UrlOy(
43                  "https://github.com/objectionary/home/tree/%s/objects/%s",
44                  "abcde"
45              ).value("org.eolang.test"),
46              Matchers.is(
47                  new URL("https://github.com/objectionary/home/tree/abcde/objects/org/eolang/test")
48              )
49          );
50      }
51  
52      @Test
53      void throwsExceptionOnInvalidUrlForProgram() {
54          Assertions.assertThrows(
55              MalformedURLException.class,
56              () -> new OyRemote.UrlOy(
57                  "hts:raw.githubusercontent.com/objectionary/home/%s/objects/%s.eo",
58                  "xyz"
59              ).value("org.eolang.app"),
60              "Expected MalformedURLException when the URL format is invalid"
61          );
62      }
63  
64      @Test
65      void throwsExceptionOnInvalidUrlForDirectory() {
66          Assertions.assertThrows(
67              MalformedURLException.class,
68              () -> new OyRemote.UrlOy(
69                  "hts:github.com/objectionary/home/tree/%s/objects/%s",
70                  "xyz"
71              ).value("org.eolang.dir"),
72              "Expected MalformedURLException when the URL format is invalid"
73          );
74      }
75  
76      @Test
77      @ExtendWith(WeAreOnline.class)
78      void checksPresenceOfProgram() throws IOException {
79          final CommitHash hash = new ChRemote("master");
80          final Objectionary objectionary = new OyRemote(hash);
81          MatcherAssert.assertThat(
82              "OyRemote positively checks the presence of the program in Objectionary",
83              objectionary.contains("org.eolang.io.stdout"),
84              Matchers.is(true)
85          );
86      }
87  
88      @Test
89      @ExtendWith(WeAreOnline.class)
90      void checksPresenceOfDirectory() throws IOException {
91          final CommitHash hash = new ChRemote("master");
92          final Objectionary objectionary = new OyRemote(hash);
93          MatcherAssert.assertThat(
94              "OyRemote positively checks the presence of the directory in Objectionary",
95              objectionary.contains("org.eolang.ms"),
96              Matchers.is(true)
97          );
98      }
99  
100     @Test
101     @ExtendWith(WeAreOnline.class)
102     void checksPresenceOfProgramWithNarrowHash() throws IOException {
103         final String stdout = "org.eolang.io.stdout";
104         MatcherAssert.assertThat(
105             String.format(
106                 "OyRemote with narrow hash should have contained program %s, but it didn't",
107                 stdout
108             ),
109             new OyRemote(
110                 new ChNarrow(
111                     new ChRemote("master")
112                 )
113             ).contains(stdout),
114             Matchers.is(true)
115         );
116     }
117 
118     @Test
119     @ExtendWith(WeAreOnline.class)
120     void checksPresenceOfDirectoryWithNarrowHash() throws IOException {
121         final String stdout = "org.eolang.ss";
122         MatcherAssert.assertThat(
123             String.format(
124                 "OyRemote with narrow hash should have contained directory %s, but it didn't",
125                 stdout
126             ),
127             new OyRemote(
128                 new ChNarrow(
129                     new ChRemote("master")
130                 )
131             ).contains(stdout),
132             Matchers.is(true)
133         );
134     }
