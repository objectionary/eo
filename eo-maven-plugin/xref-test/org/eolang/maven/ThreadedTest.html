5   package org.eolang.maven;
6   
7   import java.util.Collections;
8   import java.util.List;
9   import org.cactoos.list.ListOf;
10  import org.hamcrest.MatcherAssert;
11  import org.hamcrest.Matchers;
12  import org.junit.jupiter.api.Test;
13  
14  /**
15   * Tests for {@link Threaded}.
16   *
17   * @since 0.56.5
18   */
19  final class ThreadedTest {
20  
21      /**
22       * Logs all exceptions.
23       * @checkstyle IllegalCatchCheck (25 lines)
24       * @checkstyle MethodBodyCommentsCheck (25 lines)
25       */
26      @SuppressWarnings({"PMD.AvoidCatchingGenericException", "PMD.AvoidThrowingRawExceptionTypes"})
27      @Test
28      void logsAllExceptionsInTheLogsOnFailure() {
29          final List<String> logs = Collections.synchronizedList(new ListOf<>());
30          try {
31              new Threaded<>(
32                  new ListOf<>(1, 2, 3),
33                  input -> {
34                      throw new RuntimeException(String.format("Failure on: %d", input));
35                  },
36                  logs::add
37              ).total();
38          } catch (final Exception ignored) {
39              // Swallow the exception in order to do asserts
40          }
41          MatcherAssert.assertThat(
42              "Logs don't have all failure messages, but they should",
43              logs,
44              Matchers.hasItems(
45                  Matchers.containsString("Failed to process \"1\""),
46                  Matchers.containsString("Failed to process \"2\""),
47                  Matchers.containsString("Failed to process \"3\"")
48              )
49          );
50      }
51  }
