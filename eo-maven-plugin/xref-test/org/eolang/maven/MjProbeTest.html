19  /**
20   * Test case for {@link MjProbe}.
21   *
22   * @since 0.28.11
23   */
24  @ExtendWith(WeAreOnline.class)
25  @ExtendWith(MktmpResolver.class)
26  final class MjProbeTest {
27  
28      @Test
29      void findsProbesViaOfflineHashFile(@Mktmp final Path temp) throws IOException {
30          final String tag = "master";
31          final String tags = "org/eolang/maven/commits/tags.txt";
32          new Saved(
33              new ResourceOf(tags),
34              temp.resolve("tags.txt")
35          ).value();
36          final String expected = "11";
37          MatcherAssert.assertThat(
38              String.format(
39                  "Number of objects that we should find during the probing phase should be equal %s",
40                  expected
41              ),
42              new FakeMaven(temp)
43                  .with("hash", new ChCached(new ChText(temp.resolve("tags.txt"), tag)))
44                  .withProgram(MjProbeTest.program())
45                  .execute(new FakeMaven.Probe())
46                  .programTojo()
47                  .probed(),
48              Matchers.equalTo(expected)
49          );
50      }
51  
52      /**
53       * Finds probes in objectionary remote repository.
54       * @param temp Temporary folder
55       * @throws IOException If some problem inside
56       * @todo #4526:90min Fix flaky {@link #findsProbesInOyRemote(Path)} test.
57       *  The test sometimes fails with the following error:
58       *  MjProbeTest.findsProbesInOyRemote:62 We should find 10 objects in
59       *  git repository with tag '0.50.0', but 9 found.
60       *  This might happen because the remote repository structure changes over time.
61       *  We need to investigate this issue and fix the test to make it stable.
62       */
63      @Test
64      @Disabled
65      void findsProbesInOyRemote(@Mktmp final Path temp) throws IOException {
66          final String tag = "0.50.0";
67          final String expected = "10";
68          final String found = new FakeMaven(temp)
69              .with("tag", tag)
70              .with("objectionary", new OyRemote(new ChRemote(tag)))
71              .withProgram(MjProbeTest.program())
72              .execute(new FakeMaven.Probe())
73              .programTojo()
74              .probed();
75          MatcherAssert.assertThat(
76              String.format(
77                  "We should find %s objects in git repository with tag '%s', but %s found",
78                  expected, tag, found
79              ),
80              found,
81              Matchers.equalTo(expected)
82          );
83      }
84  
85      private static String[] program() {
86          return new String[]{
87              "+package foo.x",
88              "+also while\n",
89              "# No comments.",
90              "[] > main",
91              "  QQ.io.stdout > @",
92              "    QQ.txt.sprintf",
93              "      \"I am %d years old\"",
94              "      plus.",
95              "        1337",
96              "        228",
97          };
98      }
