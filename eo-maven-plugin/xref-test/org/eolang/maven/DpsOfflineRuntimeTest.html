5   package org.eolang.maven;
6   
7   import org.cactoos.list.ListOf;
8   import org.hamcrest.MatcherAssert;
9   import org.hamcrest.Matchers;
10  import org.junit.jupiter.api.Test;
11  
12  /**
13   * Tests for {@link DpsOfflineRuntime}.
14   *
15   * @since 0.56.5
16   */
17  final class DpsOfflineRuntimeTest {
18  
19      @Test
20      void addsCurrentEoDependency() {
21          MatcherAssert.assertThat(
22              "Offline EO runtime dependency does not match with expected",
23              new DpsOfflineRuntime(new ListOf<>()),
24              Matchers.hasItem(
25                  Matchers.hasToString(
26                      Matchers.containsString("org.eolang:eo-runtime")
27                  )
28              )
29          );
30      }
31  
32      @Test
33      void doesNotDuplicatesRuntime() {
34          MatcherAssert.assertThat(
35              "Size of dependencies does not match with expected",
36              new DpsOfflineRuntime(
37                  new ListOf<>(
38                      new Dep().withGroupId("org.eolang")
39                          .withArtifactId("eo-runtime")
40                          .withVersion("0.56.2")
41                  )
42              ),
43              Matchers.iterableWithSize(1)
44          );
45      }
46  
47      @Test
48      void usesAlreadyDefinedVersion() {
49          MatcherAssert.assertThat(
50              "Offline EO runtime dependency should use already defined version",
51              new DpsOfflineRuntime(
52                  new ListOf<>(
53                      new Dep().withGroupId("org.eolang")
54                          .withArtifactId("eo-runtime")
55                          .withVersion("0.0.1")
56                  )
57              ),
58              Matchers.hasItem(
59                  Matchers.hasToString(
60                      Matchers.containsString("org.eolang:eo-runtime:0.0.1")
61                  )
62              )
63          );
64      }
65  }
