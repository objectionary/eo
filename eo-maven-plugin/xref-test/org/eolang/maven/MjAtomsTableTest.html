5   package org.eolang.maven;
6   
7   import com.yegor256.Mktmp;
8   import com.yegor256.MktmpResolver;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import org.cactoos.io.InputOf;
12  import org.eolang.parser.EoSyntax;
13  import org.hamcrest.MatcherAssert;
14  import org.hamcrest.Matchers;
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.extension.ExtendWith;
17  
18  /**
19   * Test cases for {@link MjAtomsTable}.
20   * @since 0.57
21   */
22  @ExtendWith(MktmpResolver.class)
23  final class MjAtomsTableTest {
24  
25      @Test
26      void generatesAtomsTableFromXmir(@Mktmp final Path temp) throws Exception {
27          new Saved(
28              new EoSyntax(
29                  new InputOf(
30                      String.join(
31                          System.lineSeparator(),
32                          "+package foo",
33                          "",
34                          "# Top object.",
35                          "[] > thing",
36                          "  # Atom returning bool.",
37                          "  [x] > is-good /bool",
38                          "  # Atom returning number.",
39                          "  [] > size /number"
40                      )
41                  )
42              ).parsed().toString(),
43              temp.resolve("xmir/foo/thing.xmir")
44          ).value();
45          new FakeMaven(temp)
46              .with("atomsInputDir", temp.resolve("xmir").toFile())
47              .with("atomsOutput", temp.resolve("classes/org/eolang/atoms.csv").toFile())
48              .execute(MjAtomsTable.class);
49          MatcherAssert.assertThat(
50              "Generated CSV must contain entries for every declared atom",
51              Files.readString(temp.resolve("classes/org/eolang/atoms.csv")),
52              Matchers.allOf(
53                  Matchers.containsString("Φ.foo.thing.is-good,Φ.bool"),
54                  Matchers.containsString("Φ.foo.thing.size,Φ.number")
55              )
56          );
57      }
58  
59      @Test
60      void writesEmptyTableWhenNoXmirSources(@Mktmp final Path temp) throws Exception {
61          new FakeMaven(temp)
62              .with("atomsInputDir", temp.resolve("nothing").toFile())
63              .with("atomsOutput", temp.resolve("classes/org/eolang/atoms.csv").toFile())
64              .execute(MjAtomsTable.class);
65          MatcherAssert.assertThat(
66              "Output CSV should be empty when there are no XMIR sources",
67              Files.readString(temp.resolve("classes/org/eolang/atoms.csv")),
68              Matchers.emptyString()
69          );
70      }
71  }
