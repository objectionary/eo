30  /**
31   * Test cases for {@link MjPrint}.
32   * @since 0.33.0
33   */
34  @ExtendWith(MktmpResolver.class)
35  final class MjPrintTest {
36  
37      @Test
38      void printsSuccessfully(@Mktmp final Path temp) throws Exception {
39          final Path resources = new File(
40              "../eo-parser/src/test/resources/org/eolang/parser/print-packs/xmir"
41          ).toPath();
42          final Collection<Path> walk = new Walk(resources);
43          Assumptions.assumeTrue(!walk.isEmpty());
44          for (final Path source : walk) {
45              new Saved(new TextOf(source), temp.resolve(source)).value();
46          }
47          final Path output = temp.resolve("output");
48          final Path sources = temp.resolve(resources);
49          new FakeMaven(temp)
50              .with("printSourcesDir", sources.toFile())
51              .with("printOutputDir", output.toFile())
52              .execute(new FakeMaven.Print())
53              .result();
54          for (final Path source : walk) {
55              final String src = resources.relativize(source).toString()
56                  .replace(".xmir", ".eo");
57              MatcherAssert.assertThat(
58                  String.format(
59                      "File with name %s should have existed in output directory, but it didn't",
60                      src
61                  ),
62                  Files.exists(output.resolve(Paths.get(src))),
63                  Matchers.is(true)
64              );
65          }
66      }
67  
68      @ParameterizedTest
69      @ClasspathSource(value = "org/eolang/maven/print-packs", glob = "**.yaml")
70      void printsXmirToEo(final String pack, @Mktmp final Path temp) throws Exception {
71          final Xtory xtory = new XtSticky(new XtYaml(pack));
72          Assumptions.assumeTrue(xtory.map().get("skip") == null);
73          MatcherAssert.assertThat(
74              "PrintMojo should print EO in straight notation, but it didn't",
75              MjPrintTest.printed(xtory, temp, false).asString(),
76              Matchers.equalTo((String) xtory.map().get("printed"))
77          );
78      }
79  
80      /**
81       * Print XMIR to EO from given pack.
82       * @param xtory XaX story
83       * @param temp Temp directory
84       * @param reversed Should notation be reversed or not
85       * @return Result printed EO
86       * @throws Exception If fails to execute {@link MjPrint}
87       */
88      private static Text printed(final Xtory xtory, final Path temp, final boolean reversed)
89          throws Exception {
90          new Saved(
91              new EoSyntax(
92                  new InputOf(xtory.map().get("origin").toString())
93              ).parsed().toString(),
94              temp.resolve("xmir/foo/x/main.xmir")
95          ).value();
96          final Map<String, Path> result = new FakeMaven(temp)
97              .with("printSourcesDir", temp.resolve("xmir").toFile())
98              .with("printOutputDir", temp.resolve("eo").toFile())
99              .with("printReversed", reversed)
100             .execute(MjPrint.class)
101             .result();
102         return new TextOf(result.get("eo/foo/x/main.eo"));
103     }
