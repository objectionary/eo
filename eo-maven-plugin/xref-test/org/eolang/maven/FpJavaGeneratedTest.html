22  /**
23   * Tests for {@link FpJavaGenerated}.
24   *
25   * @since 0.56.7
26   */
27  @ExtendWith(MktmpResolver.class)
28  final class FpJavaGeneratedTest {
29  
30      @Test
31      void placesTheJavaAsInputFiles(@Mktmp final Path temp) throws Exception {
32          final Path source = temp.resolve("foo.eo");
33          Files.write(source, "# Foo.\n[] > foo".getBytes(StandardCharsets.UTF_8));
34          final Path java = temp.resolve("Foo.java");
35          final String content = "public final class Foo {}";
36          new FpJavaGenerated(
37              new Xnav(new Xembler(new Directives().add("java").set(content)).xml()),
38              new FileGenerationReport(
39                  new AtomicInteger(), new Place("foo").make(temp, MjAssemble.EO), java
40              )
41          ).apply(source, java);
42          MatcherAssert.assertThat(
43              "Placed Java code does not match with expected",
44              new TextOf(java).asString(),
45              Matchers.equalTo(content)
46          );
47      }
48  }
